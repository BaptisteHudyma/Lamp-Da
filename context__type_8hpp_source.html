254 uint8_t LMBD_INLINE set_active_mode(uint8_t value, uint8_t maxValueWrap = 255)
255 {
256 if (value + 1 > maxValueWrap)
257 value = 0;
258
259 auto manager = modeManager.get_context();
260
261 // cancel any blip
262 // TODO: this should be in quit_mode
263 manager.cancel_blip();
264
265 // if next mode is the same as current one, add a blip to help user differenciate
266 if (modeManager.activeIndex.modeIndex == value)
267 {
268 // This is not complete, there is no group check, so the blip will be visible sometimes on modes with the same
269 // modeIndex in different groups
270
271 // blip to indicate a mode change to the same mode
272 manager.blip(100);
273 }
274
275 // signal that we are quitting the mode
276 modeManager.quit_mode(manager);
277 // switch mode
278 modeManager.activeIndex.modeIndex = value;
279 // signal that we entered a new mode
280 modeManager.enter_mode(manager);
281 return value;
282 }
