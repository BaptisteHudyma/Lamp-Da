272 uint8_t LMBD_INLINE set_active_mode(uint8_t value, uint8_t maxValueWrap = 255)
273 {
274 if (value + 1 > maxValueWrap)
275 value = 0;
276
277 auto manager = modeManager.get_context();
278
279 // cancel any blip
280 // TODO: this should be in quit_mode
281 manager.cancel_blip();
282
283 // if next mode is the same as current one, add a blip to help user differenciate
284 if (modeManager.activeIndex.modeIndex == value)
285 {
286 // This is not complete, there is no group check, so the blip will be visible sometimes on modes with the same
287 // modeIndex in different groups
288
289 // blip to indicate a mode change to the same mode
290 manager.blip(100);
291 }
292
293 // signal that we are quitting the mode
294 modeManager.quit_mode(manager);
295 // switch mode
296 modeManager.activeIndex.modeIndex = value;
297 // signal that we entered a new mode
298 modeManager.enter_mode(manager);
299 return value;
300 }
