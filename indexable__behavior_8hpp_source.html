168bool button_hold_default(const uint8_t clicks, const bool isEndOfHoldEvent, const uint32_t holdDuration)
169{
170 // Handle default common behavior
171 if (default_behaviors::button_hold(clicks, isEndOfHoldEvent, holdDuration))
172 {
173 // some event is already handled
174 return true;
175 }
176
177 auto manager = get_context();
178 auto& rampHandler = manager.state.rampHandler;
179
180 switch (clicks)
181 {
182 case 4: // 4 click+hold: configure favorite
183 {
184 if (not isEndOfHoldEvent)
185 {
186 // lock to prevent addition of favorite from favorite & if we are deleting favorites
187 // the second case can happen when the user delete all favorites
188 if (not manager.state.isInFavoriteMockGroup && not manager.state.isInDeleteFavorite)
189 {
190 // no new favorite in favorite
191 manager.animate_favorite_pick(holdDuration, 1500);
192 }
193 else
194 {
195 // remove current favorite
196 manager.animate_favorite_delete(holdDuration, 2000);
197 }
198 }
199 else
200 {
201 manager.state.isInDeleteFavorite = false;
202 }
203 return true;
204 }
205
206 case 6: // 6 click+hold: scroll across modes and group
207 {
208 // no scroll in favorite
209 if (not manager.state.isInFavoriteMockGroup)
210 {
211 // register end of scroll
212 if (isEndOfHoldEvent)
213 {
214 // update release time
215 manager.state.lastScrollStopped = manager.lamp.now;
216 }
217 else
218 {
219 manager.handle_scroll_modes(holdDuration);
220 }
221 }
222 return true;
223 }
224
225 default:
226 break;
227 }
228 return false;
229}

238void handle_pattern_select_command(const uint8_t patternIndex, const uint32_t requestedColor = 0)
239{
240 // ignore if not on
241 if (not logic::behavior::is_in_output_state())
242 return;
243
244 // get manager and execute a mode or group change
245 auto manager = get_context();
246 if (manager.get_hidden_groups_count() <= 0)
247 {
248 platform::lampda_print("Cannot use bluetooth control without an hidden bluetooth group");
249 return;
250 }
251
252 const auto bluetoothGroup = manager.template get_group_id_of_mode<modes::bluetooth::ColorControlMode>();
253 if (bluetoothGroup < 0)
254 {
255 platform::lampda_print("Cannot find the bluetooth group and modes");
256 return;
257 }
258
259 const auto availableModes = manager.get_modes_count();
260 if (availableModes <= patternIndex)
261 {
262 platform::lampda_print("Unimplemented pattern id %d", patternIndex);
263 return;
264 }
265
266 manager.set_active_group(bluetoothGroup);
267 manager.set_active_mode(patternIndex);
268
269 // sanity check
270 const int trueIndex = manager.template get_mode_id<modes::bluetooth::ColorControlMode>();
271 assert(trueIndex == 0 && "Bluetooth group should have the programmable color output as the first group index");
272
273 // handle special programmable group mode
274 if (patternIndex == 0)
275 {
276 auto& modeState = manager.template get_state_of_mode<modes::bluetooth::ColorControlMode>();
277 modeState.color = requestedColor;
278 }
279}

283void handle_elk_command(const utils::ELK::Package& elkControlCommand)
284{
285 // Handle default common behavior
286 if (default_behaviors::handle_elk_command(elkControlCommand))
287 {
288 // some event is already handled
289 return;
290 }
291
292 switch (elkControlCommand.type)
293 {
294 case utils::ELK::Type::COLOR_SELECT:
295 {
296 const uint32_t color =
297 elkControlCommand.data[0] << 16 | elkControlCommand.data[1] << 8 | elkControlCommand.data[2];
298 __private_elk::handle_pattern_select_command(0, color);
299 break;
300 }
301 case utils::ELK::Type::PATTERN_SELECT:
302 {
303 __private_elk::handle_pattern_select_command(elkControlCommand.data[0] + 1);
304 break;
305 }
306 // unhandled
307 default:
308 {
309 platform::lampda_print("Unsupported ELK message type");
310 break;
311 }
312 }
313}
