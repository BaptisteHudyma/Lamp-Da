76 void update(LampTy& lamp, auto& callback)
77 {
78 auto& buffer = lamp.template getTempBuffer<dstBufIdx>();
79
80 uint16_t nextLine = (lastLine + 1) % (nbLines + 1);
81 uint16_t nextStart = nextLine * fwidth;
82 bool endReached = (nextStart + width) > buffer.size();
83
84 // if scrollAmount is 0, then we prefer wrapping the grid
85 if (endReached && ConfigTy::scrollAmount == 0)
86 {
87 nextStart = 0;
88 lastLine = (1 + lastLine) % (nbLines + 1);
89 }
90
91 // if not, scroll grid by scrollAmount (skewed or not)
92 else if (endReached)
93 {
94 scrollBy(lamp, ConfigTy::scrollAmount, ConfigTy::scrollSkewed);
95
96 lastLine -= ConfigTy::scrollAmount;
97 nextLine -= ConfigTy::scrollAmount;
98 nextStart = nextLine * fwidth;
99 }
100
101 // call callback
102 LineTy newLine {};
103 callback(currentLine, newLine);
104 currentLine = newLine;
105
106 // save result to buffer & move to next line
107 if (assertBool(nextStart + width <= buffer.size()))
108 std::copy(newLine.begin(), newLine.end(), buffer.begin() + nextStart);
109
110 lastLine = (1 + lastLine) % (nbLines + 1);
111 }

205 template<bool hasCustomRamp, uint8_t rampSubstitute = 70, uint8_t baseSmooth = 4>
206 void LMBD_INLINE loop(auto& ctx, auto& callback)
207 {
208 auto& lamp = ctx.lamp;
209 uint8_t rampValue = (hasCustomRamp ? ctx.get_active_custom_ramp() : rampSubstitute);
210
211 // by default, quickly run smoothly the grid scrolling
212 uint16_t smoothFrame = baseSmooth + rampValue / 16;
213 if (smoothFrame <= 10 && lamp.getBrightness() > 32)
214 {
215 uint32_t counter = lamp.raw_frame_count % smoothFrame;
216 if (!counter)
217 smoothUpdate(lamp, callback);
218
219 smoothDisplay(lamp, counter, smoothFrame);
220 if constexpr (ConfigTy::renderBlurAmount)
221 lamp.blur(ConfigTy::renderBlurAmount);
222
223 // coarser display for slow scrolling (smoothing too visible)
224 }
225 else
226 {
227 int now = lamp.now;
228 int frameDuration = rampValue + lamp.frameDurationMs;
229
230 if (abs(int(now - lastUpdate)) > frameDuration)
231 {
232 lastUpdate = now;
233 update(lamp, callback);
234
235 display(lamp);
236 if constexpr (ConfigTy::renderBlurAmount)
237 lamp.blur(ConfigTy::renderBlurAmount);
238 }
239 }
240 }

257template<uint8_t ruleNo, bool straight = false, bool leftBound = false, bool rightBound = false, uint8_t nbBits = 24>
258void wolframRule(const auto& before, auto& after)
259{
260 constexpr uint8_t pattern[8] = {0b1 & (ruleNo >> 7),
261 0b1 & (ruleNo >> 6),
262 0b1 & (ruleNo >> 5),
263 0b1 & (ruleNo >> 4),
264 0b1 & (ruleNo >> 3),
265 0b1 & (ruleNo >> 2),
266 0b1 & (ruleNo >> 1),
267 0b1 & (ruleNo >> 0)};
268
269 uint16_t start = leftBound ? 1 : 0;
270 uint16_t end = rightBound ? after.size() - 1 : after.size();
271
272 for (uint16_t I = start; I < end; ++I)
273 {
274 uint16_t J = (I + (straight ? 1 : 0)) % end;
275 after[J] = 0;
276
277 for (uint32_t P = 0; P < nbBits; ++P)
278 {
279 uint32_t mask = 1 << P;
280 uint32_t l = before[(I + before.size() - 1) % before.size()] & mask;
281 uint32_t m = before[I] & mask;
282 uint32_t r = before[(I + 1) % before.size()] & mask;
283
284 uint8_t lmr = 0;
285 if (l)
286 lmr |= 0b100;
287 if (m)
288 lmr |= 0b010;
289 if (r)
290 lmr |= 0b001;
291
292 if (pattern[lmr])
293 {
294 after[J] |= mask;
295 }
296 }
297 }
298}
