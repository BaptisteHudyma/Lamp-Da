28template<typename T, typename U> class LinearCurve
29{
30public:
32 using point_t = Point<T, U>;
33
35 LinearCurve(const std::vector<point_t>& points)
36 {
37 // first fail fast
38 assert(points.size() >= 2 && "Linear curve must have more than 1 points");
39
40 pts = points;
41
42 for (const auto& p: pts)
43 {
44 // check A
45 assert(not std::isnan(p.x) && "invalid value in curve parameters");
46 assert(std::isfinite(p.x) && "invalid value in curve parameters");
47 assert(not std::isnan(p.y) && "invalid value in curve parameters");
48 assert(std::isfinite(p.y) && "invalid value in curve parameters");
49 }
50
51 // sort the vector by the x coordinate
52 auto lbd = [](const point_t& a, const point_t& b) {
53 return a.x < b.x;
54 };
55 std::sort(pts.begin(), pts.end(), lbd);
56
57 // remove duplicates (after sorting)
58 std::vector<point_t> nPoints;
59 nPoints.reserve(pts.size());
60
61 point_t lastPt = pts[0];
62 nPoints.emplace_back(lastPt);
63 for (size_t i = 1; i < pts.size(); i++)
64 {
65 const auto& p = pts[i];
66 if (lastPt.x != p.x and lastPt.y != p.y)
67 nPoints.emplace_back(p);
68 lastPt = p;
69 }
70 // last check
71 pts = nPoints;
72 assert(pts.size() >= 2 && "Linear curve must have more than 1 points");
73 }
74
76 U sample(const T x) const
77 {
78 point_t lastPt = pts[0];
79 // bounds failure
80 if (std::isnan(x))
81 return lastPt.y;
82 if (x <= lastPt.x)
83 return lastPt.y;
84 if (x >= pts.back().x)
85 return pts.back().y;
86
87 for (size_t i = 1; i < pts.size(); ++i)
88 {
89 const point_t& pt = pts[i];
90 // in this segment bound
91 if (x >= lastPt.x and x <= pt.x)
92 {
93 return lmpd_map<U>(x, lastPt.x, pt.x, lastPt.y, pt.y);
94 }
95 // update last point
96 lastPt = pt;
97 }
98
99 // highest bound failure
100 return lastPt.y;
101 }
102
103private:
105 std::vector<point_t> pts;
106};

113template<typename T, typename U> class ExponentialCurve
114{
115public:
117 using point_t = Point<T, U>;

126 ExponentialCurve(const point_t& pointA, const point_t& pointB, const double exponent = 15.0) :
127 lowerBound(pointA.y),
128 upperBound(pointB.y)
129 {
130 const double div = pointA.y / static_cast<double>(pointB.y);
131 if (div <= -1.0)
132 {
133 // Error: linear approx...
134 _exp = 1;
135 _a = 0;
136 _b = 1;
137 return;
138 }
139 const double A = exp(log(div) / exponent);
140 _exp = exponent;
141 _a = (static_cast<double>(pointA.x) - static_cast<double>(pointB.x) * A) / (A - 1);
142 _b = static_cast<double>(pointA.y) / pow(static_cast<double>(pointA.x + _a), exponent);
143 }
144
150 float sample(const T x) const
151 {
152 const float res = pow(static_cast<double>(x) + _a, _exp) * _b;
153 return lmpd_constrain<float>(res, lowerBound, upperBound);
154 }
155
156private:
158 const T lowerBound;
160 const T upperBound;
161
163 double _exp;
165 double _a;
167 double _b;
168};
