33template<typename T, typename U> class LinearCurve
34{
35public:
37 using point_t = Point<T, U>;

40 LinearCurve(const std::vector<point_t>& points)
41 {
42 // first fail fast
43 assert(points.size() >= 2 && "Linear curve must have more than 1 points");
44
45 pts = points;
46
47 for (const auto& p: pts)
48 {
49 // check A
50 assert(not std::isnan(p.x) && "invalid value in curve parameters");
51 assert(std::isfinite(p.x) && "invalid value in curve parameters");
52 assert(not std::isnan(p.y) && "invalid value in curve parameters");
53 assert(std::isfinite(p.y) && "invalid value in curve parameters");
54 }
55
56 // sort the vector by the x coordinate
57 auto lbd = [](const point_t& a, const point_t& b) {
58 return a.x < b.x;
59 };
60 std::sort(pts.begin(), pts.end(), lbd);
61
62 // remove duplicates (after sorting)
63 std::vector<point_t> nPoints;
64 nPoints.reserve(pts.size());
65
66 point_t lastPt = pts[0];
67 nPoints.emplace_back(lastPt);
68 for (size_t i = 1; i < pts.size(); i++)
69 {
70 const auto& p = pts[i];
71 if (lastPt.x != p.x and lastPt.y != p.y)
72 nPoints.emplace_back(p);
73 lastPt = p;
74 }
75 // last check
76 pts = nPoints;
77 assert(pts.size() >= 2 && "Linear curve must have more than 1 points");
78 }
79
81 U sample(const T x) const
82 {
83 point_t lastPt = pts[0];
84 // bounds failure
85 if (std::isnan(x))
86 return lastPt.y;
87 if (x <= lastPt.x)
88 return lastPt.y;
89 if (x >= pts.back().x)
90 return pts.back().y;
91
92 for (size_t i = 1; i < pts.size(); ++i)
93 {
94 const point_t& pt = pts[i];
95 // in this segment bound
96 if (x >= lastPt.x and x <= pt.x)
97 {
98 return lmpd_map<U>(x, lastPt.x, pt.x, lastPt.y, pt.y);
99 }
100 // update last point
101 lastPt = pt;
102 }
103
104 // highest bound failure
105 return lastPt.y;
106 }
107
108private:
110 std::vector<point_t> pts;
111};

131 ExponentialCurve(const point_t& pointA, const point_t& pointB, const double exponent = 15.0) :
132 lowerBound(pointA.y),
133 upperBound(pointB.y)
134 {
135 const double div = pointA.y / static_cast<double>(pointB.y);
136 if (div <= -1.0)
137 {
138 // Error: linear approx...
139 _exp = 1;
140 _a = 0;
141 _b = 1;
142 return;
143 }
144 const double A = exp(log(div) / exponent);
145 _exp = exponent;
146 _a = (static_cast<double>(pointA.x) - static_cast<double>(pointB.x) * A) / (A - 1);
147 _b = static_cast<double>(pointA.y) / pow(static_cast<double>(pointA.x + _a), exponent);
148 }
149
155 float sample(const T x) const
156 {
157 const float res = pow(static_cast<double>(x) + _a, _exp) * _b;
158 return lmpd_constrain<float>(res, lowerBound, upperBound);
159 }
160
161private:
163 const T lowerBound;
165 const T upperBound;
166
168 double _exp;
170 double _a;
172 double _b;
173};
