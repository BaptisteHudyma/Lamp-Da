456 {

550 void LMBD_INLINE setBrightness(const brightness_t newBrightness,
551 const bool skipCallbacks = true,
552 const bool skipUpdateBrightness = false,
553 const bool updateSavedBrightess = false)
554 {
555 assert((skipCallbacks || !skipUpdateBrightness) && "implicit callback skip!");
556
557 if (!skipUpdateBrightness)
558 {
559 uint32_t last = utils::brightness::when_last_update_brightness();
560 if ((this->now - last) > brightnessDurationMs)
561 utils::brightness::update_brightness(newBrightness, skipCallbacks);
562 }
563 if (updateSavedBrightess)
564 utils::brightness::update_saved_brightness();
565
566 // USE THE BRIGHTNESS VALUE AFTER THE UPDATE
567 // brightness can be limited by the system, so do not use the raw brightness
568 const auto trueNewBrightness = utils::brightness::get_brightness();
569 const auto trueMaxBrightness = utils::brightness::get_max_brightness();
570
571 if constexpr (flavor == LampTypes::indexable)
572 {
573 constexpr uint8_t minBrightness = 5;
574 using curve_t = utils::curves::LinearCurve<brightness_t, uint8_t>;
575 static curve_t brightnessCurve(
576 {curve_t::point_t {0, minBrightness}, curve_t::point_t {brightness::absoluteMaximumBrightness, 255}});
577
578 strip.setBrightness(brightnessCurve.sample(trueNewBrightness));
579 }
580
581 if constexpr (flavor == LampTypes::simple)
582 {
583 if (trueNewBrightness >= trueMaxBrightness)
584 physical::outputPower::blip(50); // blip
585
586 constexpr uint16_t maxOutputVoltage_mV = stripInputVoltage_mV;
587 using curve_t = utils::curves::ExponentialCurve<brightness_t, uint16_t>;
588 static curve_t brightnessCurve(curve_t::point_t {0, 9400},
589 curve_t::point_t {brightness::absoluteMaximumBrightness, maxOutputVoltage_mV},
590 1.0);
591
592 physical::outputPower::write_voltage(round(brightnessCurve.sample(trueNewBrightness)));
593 }

706 void LMBD_INLINE blur(uint8_t blurBy)
707 {
708 if (blurBy == 0)
709 return; // optimization: 0 means "don't blur"
710 uint8_t keep = 255 - blurBy;
711 uint8_t seep = blurBy >> 1;
712
713 uint32_t carryover = 0;
714 for (unsigned i = 0; i < ledCount; i++)
715 {
716 uint32_t cur = strip.getPixelColor(i);
717 uint32_t c = cur;
718 uint32_t part = colors::fade<false>(c, seep);
719 cur = colors::add<true>(colors::fade<false>(c, keep), carryover);
720 if (i > 0)
721 {
722 c = strip.getPixelColor(i - 1);
723 setPixelColor(i - 1, colors::add<true>(c, part));
724 }
725 setPixelColor(i, cur);
726 carryover = part;
727 }
728 }

1040static constexpr uint16_t to_strip(uint16_t x, uint16_t y)
1041{
1042 // maxHeight is the last "full" row and maxOverflowHeight is truncated row
1043 // -> user max use to_strip(x, maxOverflowHeight) to set truncated row
1044 if (y >= hardware::LampTy::maxOverflowHeight)
1045 {
1046 y = hardware::LampTy::maxOverflowHeight - 1;
1047 }
1048
1049 // if row is longer (because next row gets a shift) use maxOverflowWidth
1050 if (hardware::LampTy::allDeltaResiduesY[y])
1051 {
1052 if (x >= hardware::LampTy::maxOverflowWidth)
1053 {
1054 x = hardware::LampTy::maxOverflowWidth - 1;
1055 }
1056
1057 // if row is aligned to XY maxWidth/maxHeight matrix, use maxWidth instead
1058 }
1059 else if (x >= hardware::LampTy::maxWidth)
1060 {
1061 x = hardware::LampTy::maxWidth - 1;
1062 }
1063
1064 // final result is capped by the "real" LED count of the strip
1065 uint16_t n = x + y * hardware::LampTy::maxWidth + hardware::LampTy::allResiduesY[y];
1066 if (n >= hardware::LampTy::ledCount)
1067 n = hardware::LampTy::ledCount - 1;
1068 return n;
1069}

1148static constexpr uint16_t to_led_index(const float angle_rad, const float z_mm)
1149{
1150 constexpr uint16_t maxZCoordinate =
1151 floor(-to_helix_z(hardware::LampTy::ledCount) / hardware::LampTy::ledStripWidth_mm);
1152
1153 // snip Z per possible lines
1154 uint16_t zIndex = floor(-z_mm / hardware::LampTy::ledStripWidth_mm);
1155 if (zIndex < 0.0)
1156 zIndex = 0.0;
1157 if (zIndex > maxZCoordinate)
1158 zIndex = maxZCoordinate;
1159
1160 // indexing around the led turn
1161 const float angularPosition = wrap_angle(angle_rad) / c_TWO_PI * hardware::LampTy::maxWidthFloat;
1162
1163 // convert to led index (approx)
1164 int16_t ledIndex = round(angularPosition + zIndex * hardware::LampTy::maxWidthFloat);
1165 if (ledIndex < 0)
1166 ledIndex = round(angularPosition + (zIndex + 1) * hardware::LampTy::maxWidthFloat);
1167 if (ledIndex >= hardware::LampTy::ledCount)
1168 ledIndex = round(angularPosition + (zIndex - 1) * hardware::LampTy::maxWidthFloat);
1169
1170 if (ledIndex < 0)
1171 return 0;
1172 if (ledIndex >= hardware::LampTy::ledCount)
1173 return hardware::LampTy::ledCount - 1;
1174 return ledIndex;
1175}
