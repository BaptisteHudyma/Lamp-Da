451 template<typename T> static constexpr LMBD_INLINE uint16_t toLedCount(T scale)
452 {
453 if constexpr (std::is_same_v<T, float>)
454 {
455 assert(scale <= 1.0 && "scale is 0-1");
456 return ledCount * scale;
457 }
458 else if constexpr (std::is_same_v<T, uint8_t>)
459 {
460 uint32_t scaled = ledCount * scale;
461 return scaled / 256;
462 }
463 else
464 {
465 static_assert(std::is_same_v<T, std::false_type>, "Invalid type, either float or uint8_t required!");
466 return 0;
467 }

546 void LMBD_INLINE setBrightness(const brightness_t newBrightness,
547 const bool skipCallbacks = true,
548 const bool skipUpdateBrightness = false,
549 const bool updateSavedBrightess = false)
550 {
551 assert((skipCallbacks || !skipUpdateBrightness) && "implicit callback skip!");
552
553 if (!skipUpdateBrightness)
554 {
555 uint32_t last = brightness::when_last_update_brightness();
556 if ((this->now - last) > brightnessDurationMs)
557 brightness::update_brightness(newBrightness, skipCallbacks);
558 }
559 if (updateSavedBrightess)
560 brightness::update_saved_brightness();
561
562 // USE THE BRIGHTNESS VALUE AFTER THE UPDATE
563 // brightness can be limited by the system, so do not use the raw brightness
564 const auto trueNewBrightness = brightness::get_brightness();
565 const auto trueMaxBrightness = brightness::get_max_brightness();
566
567 if constexpr (flavor == LampTypes::indexable)
568 {
569 constexpr uint8_t minBrightness = 5;
570 using curve_t = curves::LinearCurve<brightness_t, uint8_t>;
571 static curve_t brightnessCurve(
572 {curve_t::point_t {0, minBrightness}, curve_t::point_t {brightness::absoluteMaximumBrightness, 255}});
573
574 strip.setBrightness(brightnessCurve.sample(trueNewBrightness));
575 }
576
577 if constexpr (flavor == LampTypes::simple)
578 {
579 if (trueNewBrightness >= trueMaxBrightness)
580 outputPower::blip(50); // blip
581
582 constexpr uint16_t maxOutputVoltage_mV = stripInputVoltage_mV;
583 using curve_t = curves::ExponentialCurve<brightness_t, uint16_t>;
584 static curve_t brightnessCurve(curve_t::point_t {0, 9400},
585 curve_t::point_t {brightness::absoluteMaximumBrightness, maxOutputVoltage_mV},
586 1.0);
587
588 outputPower::write_voltage(round(brightnessCurve.sample(trueNewBrightness)));
589 }
590 }

702 void LMBD_INLINE blur(uint8_t blurBy)
703 {
704 if (blurBy == 0)
705 return; // optimization: 0 means "don't blur"
706 uint8_t keep = 255 - blurBy;
707 uint8_t seep = blurBy >> 1;
708
709 uint32_t carryover = 0;
710 for (unsigned i = 0; i < ledCount; i++)
711 {
712 uint32_t cur = strip.getPixelColor(i);
713 uint32_t c = cur;
714 uint32_t part = colors::fade<false>(c, seep);
715 cur = colors::add<true>(colors::fade<false>(c, keep), carryover);
716 if (i > 0)
717 {
718 c = strip.getPixelColor(i - 1);
719 setPixelColor(i - 1, colors::add<true>(c, part));
720 }
721 setPixelColor(i, cur);
722 carryover = part;
723 }
724 }

1033static constexpr uint16_t to_strip(uint16_t x, uint16_t y)
1034{
1035 // maxHeight is the last "full" row and maxOverflowHeight is truncated row
1036 // -> user max use to_strip(x, maxOverflowHeight) to set truncated row
1037 if (y >= hardware::LampTy::maxOverflowHeight)
1038 {
1039 y = hardware::LampTy::maxOverflowHeight - 1;
1040 }
1041
1042 // if row is longer (because next row gets a shift) use maxOverflowWidth
1043 if (hardware::LampTy::allDeltaResiduesY[y])
1044 {
1045 if (x >= hardware::LampTy::maxOverflowWidth)
1046 {
1047 x = hardware::LampTy::maxOverflowWidth - 1;
1048 }
1049
1050 // if row is aligned to XY maxWidth/maxHeight matrix, use maxWidth instead
1051 }
1052 else if (x >= hardware::LampTy::maxWidth)
1053 {
1054 x = hardware::LampTy::maxWidth - 1;
1055 }
1056
1057 // final result is capped by the "real" LED count of the strip
1058 uint16_t n = x + y * hardware::LampTy::maxWidth + hardware::LampTy::allResiduesY[y];
1059 if (n >= hardware::LampTy::ledCount)
1060 n = hardware::LampTy::ledCount - 1;
1061 return n;
1062}

1065static constexpr XYTy strip_to_XY(uint16_t n)
1066{
1067 // (saturates N to ledCount)
1068 if (n >= hardware::LampTy::ledCount)
1069 n = hardware::LampTy::ledCount - 1;
1070
1071 uint16_t y = 0;
1072 while (y * hardware::LampTy::maxWidth + hardware::LampTy::allResiduesY[y] <= n)
1073 y += 1;
1074
1075 if (y == 0)
1076 return {0, 0};
1077 y -= 1;
1078

1141static constexpr uint16_t to_led_index(const float angle_rad, const float z_mm)
1142{
1143 constexpr uint16_t maxZCoordinate =
1144 floor(-to_helix_z(hardware::LampTy::ledCount) / hardware::LampTy::ledStripWidth_mm);
1145
1146 // snip Z per possible lines
1147 uint16_t zIndex = floor(-z_mm / hardware::LampTy::ledStripWidth_mm);
1148 if (zIndex < 0.0)
1149 zIndex = 0.0;
1150 if (zIndex > maxZCoordinate)
1151 zIndex = maxZCoordinate;
1152
1153 // indexing around the led turn
1154 const float angularPosition = wrap_angle(angle_rad) / c_TWO_PI * hardware::LampTy::maxWidthFloat;
1155
1156 // convert to led index (approx)
1157 int16_t ledIndex = round(angularPosition + zIndex * hardware::LampTy::maxWidthFloat);
1158 if (ledIndex < 0)
1159 ledIndex = round(angularPosition + (zIndex + 1) * hardware::LampTy::maxWidthFloat);
1160 if (ledIndex >= hardware::LampTy::ledCount)
1161 ledIndex = round(angularPosition + (zIndex - 1) * hardware::LampTy::maxWidthFloat);
1162
1163 if (ledIndex < 0)
1164 return 0;
1165 if (ledIndex >= hardware::LampTy::ledCount)
1166 return hardware::LampTy::ledCount - 1;
1167 return ledIndex;
1168}
