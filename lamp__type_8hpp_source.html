550 void LMBD_INLINE setBrightness(const brightness_t newBrightness,
551 const bool skipCallbacks = true,
552 const bool skipUpdateBrightness = false,
553 const bool updateSavedBrightess = false)
554 {
555 assert((skipCallbacks || !skipUpdateBrightness) && "implicit callback skip!");
556
557 if (!skipUpdateBrightness)
558 {
559 uint32_t last = brightness::when_last_update_brightness();
560 if ((this->now - last) > brightnessDurationMs)
561 brightness::update_brightness(newBrightness, skipCallbacks);
562 }
563 if (updateSavedBrightess)
564 brightness::update_saved_brightness();
565
566 // USE THE BRIGHTNESS VALUE AFTER THE UPDATE
567 // brightness can be limited by the system, so do not use the raw brightness
568 const auto trueNewBrightness = brightness::get_brightness();
569 const auto trueMaxBrightness = brightness::get_max_brightness();
570
571 if constexpr (flavor == LampTypes::indexable)
572 {
573 constexpr uint8_t minBrightness = 5;
574 using curve_t = curves::LinearCurve<brightness_t, uint8_t>;
575 static curve_t brightnessCurve(
576 {curve_t::point_t {0, minBrightness}, curve_t::point_t {brightness::absoluteMaximumBrightness, 255}});
577
578 strip.setBrightness(brightnessCurve.sample(trueNewBrightness));
579 }
580
581 if constexpr (flavor == LampTypes::simple)
582 {
583 if (trueNewBrightness >= trueMaxBrightness)
584 outputPower::blip(50); // blip
585
586 constexpr uint16_t maxOutputVoltage_mV = stripInputVoltage_mV;
587 using curve_t = curves::ExponentialCurve<brightness_t, uint16_t>;
588 static curve_t brightnessCurve(curve_t::point_t {0, 9400},
589 curve_t::point_t {brightness::absoluteMaximumBrightness, maxOutputVoltage_mV},
590 1.0);
591
592 outputPower::write_voltage(round(brightnessCurve.sample(trueNewBrightness)));

706 void LMBD_INLINE blur(uint8_t blurBy)
707 {
708 if (blurBy == 0)
709 return; // optimization: 0 means "don't blur"
710 uint8_t keep = 255 - blurBy;
711 uint8_t seep = blurBy >> 1;
712
713 uint32_t carryover = 0;
714 for (unsigned i = 0; i < ledCount; i++)
715 {
716 uint32_t cur = strip.getPixelColor(i);
717 uint32_t c = cur;
718 uint32_t part = colors::fade<false>(c, seep);
719 cur = colors::add<true>(colors::fade<false>(c, keep), carryover);
720 if (i > 0)
721 {
722 c = strip.getPixelColor(i - 1);
723 setPixelColor(i - 1, colors::add<true>(c, part));
724 }
725 setPixelColor(i, cur);
726 carryover = part;
727 }
728 }

1037static constexpr uint16_t to_strip(uint16_t x, uint16_t y)
1038{
1039 // maxHeight is the last "full" row and maxOverflowHeight is truncated row
1040 // -> user max use to_strip(x, maxOverflowHeight) to set truncated row
1041 if (y >= hardware::LampTy::maxOverflowHeight)
1042 {
1043 y = hardware::LampTy::maxOverflowHeight - 1;
1044 }
1045
1046 // if row is longer (because next row gets a shift) use maxOverflowWidth
1047 if (hardware::LampTy::allDeltaResiduesY[y])
1048 {
1049 if (x >= hardware::LampTy::maxOverflowWidth)
1050 {
1051 x = hardware::LampTy::maxOverflowWidth - 1;
1052 }
1053
1054 // if row is aligned to XY maxWidth/maxHeight matrix, use maxWidth instead
1055 }
1056 else if (x >= hardware::LampTy::maxWidth)
1057 {
1058 x = hardware::LampTy::maxWidth - 1;
1059 }
1060
1061 // final result is capped by the "real" LED count of the strip
1062 uint16_t n = x + y * hardware::LampTy::maxWidth + hardware::LampTy::allResiduesY[y];
1063 if (n >= hardware::LampTy::ledCount)
1064 n = hardware::LampTy::ledCount - 1;
1065 return n;
1066}

1069static constexpr XYTy strip_to_XY(uint16_t n)
1070{
1071 // (saturates N to ledCount)
1072 if (n >= hardware::LampTy::ledCount)
1073 n = hardware::LampTy::ledCount - 1;
1074
1075 uint16_t y = 0;
1076 while (y * hardware::LampTy::maxWidth + hardware::LampTy::allResiduesY[y] <= n)
1077 y += 1;
1078
1079 if (y == 0)
1080 return {0, 0};
1081 y -= 1;

1145static constexpr uint16_t to_led_index(const float angle_rad, const float z_mm)
1146{
1147 constexpr uint16_t maxZCoordinate =
1148 floor(-to_helix_z(hardware::LampTy::ledCount) / hardware::LampTy::ledStripWidth_mm);
1149
1150 // snip Z per possible lines
1151 uint16_t zIndex = floor(-z_mm / hardware::LampTy::ledStripWidth_mm);
1152 if (zIndex < 0.0)
1153 zIndex = 0.0;
1154 if (zIndex > maxZCoordinate)
1155 zIndex = maxZCoordinate;
1156
1157 // indexing around the led turn
1158 const float angularPosition = wrap_angle(angle_rad) / c_TWO_PI * hardware::LampTy::maxWidthFloat;
1159
1160 // convert to led index (approx)
1161 int16_t ledIndex = round(angularPosition + zIndex * hardware::LampTy::maxWidthFloat);
1162 if (ledIndex < 0)
1163 ledIndex = round(angularPosition + (zIndex + 1) * hardware::LampTy::maxWidthFloat);
1164 if (ledIndex >= hardware::LampTy::ledCount)
1165 ledIndex = round(angularPosition + (zIndex - 1) * hardware::LampTy::maxWidthFloat);
1166
1167 if (ledIndex < 0)
1168 return 0;
1169 if (ledIndex >= hardware::LampTy::ledCount)
1170 return hardware::LampTy::ledCount - 1;
1171 return ledIndex;
1172}
