454 template<typename T> static constexpr LMBD_INLINE uint16_t toLedCount(T scale)
455 {
456 if constexpr (std::is_same_v<T, float>)
457 {
458 assert(scale <= 1.0 && "scale is 0-1");
459 return ledCount * scale;
460 }

464 return scaled / 256;
465 }

552 void LMBD_INLINE setBrightness(const brightness_t newBrightness,
553 const bool skipCallbacks = true,
554 const bool skipUpdateBrightness = false,
555 const bool updateSavedBrightess = false)
556 {
557 assert((skipCallbacks || !skipUpdateBrightness) && "implicit callback skip!");
558
559 if (!skipUpdateBrightness)
560 {
561 uint32_t last = logic::brightness::when_last_update_brightness();
562 if ((this->now - last) > brightnessDurationMs)
563 logic::brightness::update_brightness(newBrightness, skipCallbacks);
564 }
565 if (updateSavedBrightess)
566 logic::brightness::update_saved_brightness();
567
568 // USE THE BRIGHTNESS VALUE AFTER THE UPDATE
569 // brightness can be limited by the system, so do not use the raw brightness
570 const auto trueNewBrightness = logic::brightness::get_brightness();
571 const auto trueMaxBrightness = logic::brightness::get_max_brightness();
572
573 if constexpr (flavor == LampTypes::indexable)
574 {
575 constexpr uint8_t minBrightness = 5;
576 using curve_t = utils::curves::LinearCurve<brightness_t, uint8_t>;
577 static curve_t brightnessCurve({curve_t::point_t {0, minBrightness},
578 curve_t::point_t {::lampda::brightness::absoluteMaximumBrightness, 255}});
579
580 strip.setBrightness(brightnessCurve.sample(trueNewBrightness));
581 }
582
583 if constexpr (flavor == LampTypes::simple)
584 {
585 if (trueNewBrightness >= trueMaxBrightness)
586 physical::outputPower::blip(50); // blip
587
588 constexpr uint16_t maxOutputVoltage_mV = stripInputVoltage_mV;
589 using curve_t = utils::curves::ExponentialCurve<brightness_t, uint16_t>;
590 static curve_t brightnessCurve(
591 curve_t::point_t {0, 9400},
592 curve_t::point_t {::lampda::brightness::absoluteMaximumBrightness, maxOutputVoltage_mV},
593 1.0);
594
595 physical::outputPower::write_voltage(round(brightnessCurve.sample(trueNewBrightness)));
596 }
597 }

709 void LMBD_INLINE blur(uint8_t blurBy)
710 {
711 if (blurBy == 0)
712 return; // optimization: 0 means "don't blur"
713 uint8_t keep = 255 - blurBy;
714 uint8_t seep = blurBy >> 1;
715
716 uint32_t carryover = 0;
717 for (unsigned i = 0; i < ledCount; i++)
718 {
719 uint32_t cur = strip.getPixelColor(i);
720 uint32_t c = cur;
721 uint32_t part = colors::fade<false>(c, seep);
722 cur = colors::add<true>(colors::fade<false>(c, keep), carryover);
723 if (i > 0)
724 {
725 c = strip.getPixelColor(i - 1);
726 setPixelColor(i - 1, colors::add<true>(c, part));
727 }
728 setPixelColor(i, cur);
729 carryover = part;
730 }
731 }

1051static constexpr uint16_t to_strip(uint16_t x, uint16_t y)
1052{
1053 // maxHeight is the last "full" row and maxOverflowHeight is truncated row
1054 // -> user max use to_strip(x, maxOverflowHeight) to set truncated row
1055 if (y >= hardware::LampTy::maxOverflowHeight)
1056 {
1057 y = hardware::LampTy::maxOverflowHeight - 1;
1058 }
1059
1060 // if row is longer (because next row gets a shift) use maxOverflowWidth
1061 if (hardware::LampTy::allDeltaResiduesY[y])
1062 {
1063 if (x >= hardware::LampTy::maxOverflowWidth)
1064 {
1065 x = hardware::LampTy::maxOverflowWidth - 1;
1066 }
1067
1068 // if row is aligned to XY maxWidth/maxHeight matrix, use maxWidth instead
1069 }
1070 else if (x >= hardware::LampTy::maxWidth)
1071 {
1072 x = hardware::LampTy::maxWidth - 1;
1073 }

1159static constexpr uint16_t to_led_index(const float angle_rad, const float z_mm)
1160{
1161 constexpr uint16_t maxZCoordinate =
1162 floor(-to_helix_z(hardware::LampTy::ledCount) / hardware::LampTy::ledStripWidth_mm);
1163
1164 // snip Z per possible lines
1165 uint16_t zIndex = floor(-z_mm / hardware::LampTy::ledStripWidth_mm);
1166 if (zIndex < 0.0)
1167 zIndex = 0.0;
1168 if (zIndex > maxZCoordinate)
1169 zIndex = maxZCoordinate;
1170
1171 // indexing around the led turn
1172 const float angularPosition = wrap_angle(angle_rad) / c_TWO_PI * hardware::LampTy::maxWidthFloat;
1173
1174 // convert to led index (approx)
1175 int16_t ledIndex = round(angularPosition + zIndex * hardware::LampTy::maxWidthFloat);
1176 if (ledIndex < 0)
1177 ledIndex = round(angularPosition + (zIndex + 1) * hardware::LampTy::maxWidthFloat);
1178 if (ledIndex >= hardware::LampTy::ledCount)
1179 ledIndex = round(angularPosition + (zIndex - 1) * hardware::LampTy::maxWidthFloat);
1180
1181 if (ledIndex < 0)
1182 return 0;
1183 if (ledIndex >= hardware::LampTy::ledCount)
1184 return hardware::LampTy::ledCount - 1;
1185 return ledIndex;
1186}
