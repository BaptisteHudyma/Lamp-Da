58 static void loop(auto& ctx)
59 {
60 auto& state = ctx.state;
61 state.imuEvent.update(ctx);
62
63 auto& particleSystem = state.imuEvent.particuleSystem;
64
65 // multiply by 2 because we rain on both sides of the lamp, half of the drops are not seen
66 const float expectedDropPerSecond =
67 ctx.state.rainDensity / 255.0 * heavyRainDropsPerSecond + lightRainDropsPerSecond;
68 ctx.state.rainDropSpawn += expectedDropPerSecond * ctx.lamp.frameDurationMs / 1000.0;
69
70 if (ctx.state.rainDropSpawn > 1.0)
71 {
72 // initialize particules in a deffered way, when free spots are available
73 particleSystem.init_deferred_particules(2 * ctx.state.rainDropSpawn, generate_particule_at_extremes);
74 ctx.state.rainDropSpawn = 0.0;
75 }
76
77 // no collisions between particles, and with no lamp limits
78 static constexpr bool shouldKeepInLampBounds = false;
79 particleSystem.iterate_no_collisions(
80 state.imuEvent.lastReading.accel, ctx.lamp.frameDurationMs / 1000.0, shouldKeepInLampBounds);
81 // depop particules that fell too far
82 const uint16_t activeParticles = particleSystem.depop_particules(recycle_particules_if_too_far);
83
84 ctx.lamp.fadeToBlackBy(255 - ctx.state.persistance);
85
86 // break palette size to display more colors per drops

88 const uint16_t wrapIndex = (n * 5) / static_cast<float>(activeParticles) * 255;
89 return colors::from_palette(static_cast<uint8_t>(wrapIndex % 255), ctx.state.palette);
