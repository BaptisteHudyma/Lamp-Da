100static uint32_t blend(uint32_t leftColor, uint32_t rightColor, uint16_t blend, bool b16 = false)
101{
102 if (blend == 0)
103 return leftColor;
104 uint16_t blendmax = b16 ? 0xFFFF : 0xFF;
105 if (blend >= blendmax)
106 return rightColor;
107 uint8_t shift = b16 ? 16 : 8;
108
109 ToRGB left_rgb(leftColor);
110 ToRGB right_rgb(rightColor);
111
112 return fromRGB(((right_rgb.r * blend) + (left_rgb.r * (blendmax - blend))) >> shift,
113 ((right_rgb.g * blend) + (left_rgb.g * (blendmax - blend))) >> shift,
114 ((right_rgb.b * blend) + (left_rgb.b * (blendmax - blend))) >> shift);
115}

147template<bool isFast = false> uint32_t add(uint32_t c1, uint32_t c2)
148{
149 ToRGB input1_rgb(c1);
150 ToRGB input2_rgb(c2);
151
152 if (isFast)
153 {
154 return fromRGB(
155 qadd8(input1_rgb.r, input2_rgb.r), qadd8(input1_rgb.g, input2_rgb.g), qadd8(input1_rgb.b, input2_rgb.b));
156 }
157 else
158 {
159 uint32_t r = input1_rgb.r + input2_rgb.r;
160 uint32_t g = input1_rgb.g + input2_rgb.g;
161 uint32_t b = input1_rgb.b + input2_rgb.b;
162 uint16_t max = r;
163 if (g > max)
164 max = g;
165 if (b > max)
166 max = b;
167 if (max < 256)
168 {
169 return fromRGB(r, g, b);
170 }
171 else
172 {
173 return fromRGB(r * 255 / max, g * 255 / max, b * 255 / max);
174 }
