465template<uint32_t startColor, uint32_t endColor> static constexpr PaletteTy _gradient_palette_from_color()
466{
467 using ToRGB = modes::colors::ToRGB;
468 constexpr ToRGB start {startColor};
469 constexpr ToRGB end {endColor};
470
471 PaletteTy palette {};
472 constexpr float step = palette.size() - 1;
473 constexpr float dr = (end.r - start.r) / step;
474 constexpr float dg = (end.g - start.g) / step;
475 constexpr float db = (end.b - start.b) / step;
476
477 for (size_t I = 0; I < palette.size(); ++I)
478 {
479 uint8_t r = start.r + (I * dr);
480 uint8_t g = start.g + (I * dg);
481 uint8_t b = start.b + (I * db);
482 palette[I] = modes::colors::fromRGB(r, g, b);
483 }
484
485 return palette;
486}

498template<bool PaletteLoops = true, typename UIntTy = uint8_t>
499static constexpr uint32_t from_palette(UIntTy index, const PaletteTy& palette, uint8_t brightness = 255)
500{
501 // convert to [0; 15] (divide by 16)
502 uint8_t renormIndex = index >> 4;
503 // get least significant part for the blend factor (15 levels + 0)
504 float blendIndex = (index & 0x0F) / 16.0f;
505
506 // support for uint16_t
507 if constexpr (sizeof(UIntTy) > 1)
508 {
509 const float remapedIndex = (index / static_cast<float>(UINT16_MAX)) * 16.f;
510 renormIndex = min<uint8_t>(floorf(remapedIndex), 15);
511 blendIndex = remapedIndex - renormIndex;
512 static_assert(std::is_same_v<UIntTy, uint16_t>, "u8 or u16 allowed only");
513 }
514 else
515 {
516 static_assert(std::is_same_v<UIntTy, uint8_t>, "u8 or u16 allowed only");
517 }
518
519 if (renormIndex >= 16)
520 return 0;
521
522 // limit blend index
523 if (blendIndex < 0.0f)
524 blendIndex = 0.0f;
525 if (blendIndex > 1.0f)
526 blendIndex = 1.0f;
527 blendIndex = 1.0f - blendIndex;
528
529 const uint32_t entry = palette[renormIndex];
530
531 // convert to rgb
532 uint8_t red1 = (entry & 0xff0000) >> 16;
533 uint8_t green1 = (entry & 0x00ff00) >> 8;
534 uint8_t blue1 = entry & 0x0000ff;
535
536 // need to blend the palette
537 if (blendIndex > 0)
538 {
539 uint32_t nextColor = 0;
540 if (renormIndex == 15)
541 {
542 nextColor = PaletteLoops ? palette[0] : palette[15];
543 }
544 else
545 {
546 nextColor = palette[1 + renormIndex];
547 }
548
549 const float f1 = blendIndex;
550 const float f2 = 1.0 - blendIndex;
551
552 const uint8_t red2 = (nextColor >> 16) & 0x0000ff;
553 red1 = red1 * f1;
554 red1 += red2 * f2;
555
556 const uint8_t green2 = (nextColor >> 8) & 0x0000ff;
557 green1 = green1 * f1;
558 green1 += green2 * f2;
559
560 const uint8_t blue2 = nextColor & 0x0000ff;
561 blue1 = blue1 * f1;
562 blue1 += blue2 * f2;
563 }
564
565 if (brightness != 255)
566 {
567 if (brightness != 0)
568 {
569 const float adjustedBrightness = (brightness + 1) / 256.0; // adjust for rounding
570 // Now, since brightness is nonzero, we don't need the full scale8_video
571 // logic; we can just to scale8 and then add one (unless scale8 fixed) to
572 // all nonzero inputs.
573 if (red1)
574 {
575 red1 = red1 * adjustedBrightness;
576 ++red1;
577 }
578 if (green1)
579 {
580 green1 = green1 * adjustedBrightness;
581 ++green1;
582 }
583 if (blue1)
584 {
585 blue1 = blue1 * adjustedBrightness;
586 ++blue1;
587 }
588 }
589 else
590 {
591 red1 = 0;
592 green1 = 0;
593 blue1 = 0;
594 }
595 }
596
597 // return color code
598 uint32_t outputColor = red1;
599 outputColor = (outputColor << 8) | green1;
600 outputColor = (outputColor << 8) | blue1;
601 return outputColor;
602}
