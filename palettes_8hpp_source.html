467template<uint32_t startColor, uint32_t endColor> static constexpr PaletteTy _gradient_palette_from_color()
468{
469 using ToRGB = modes::colors::ToRGB;
470 constexpr ToRGB start {startColor};
471 constexpr ToRGB end {endColor};
472
473 PaletteTy palette {};
474 constexpr float step = palette.size() - 1;
475 constexpr float dr = (end.r - start.r) / step;
476 constexpr float dg = (end.g - start.g) / step;
477 constexpr float db = (end.b - start.b) / step;
478
479 for (size_t I = 0; I < palette.size(); ++I)
480 {
481 uint8_t r = start.r + (I * dr);
482 uint8_t g = start.g + (I * dg);
483 uint8_t b = start.b + (I * db);
484 palette[I] = modes::colors::fromRGB(r, g, b);
485 }
486
487 return palette;
488}

500template<bool PaletteLoops = true, typename UIntTy = uint8_t>
501static constexpr uint32_t from_palette(UIntTy index, const PaletteTy& palette, uint8_t brightness = 255)
502{
503 // convert to [0; 15] (divide by 16)
504 uint8_t renormIndex = index >> 4;
505 // get least significant part for the blend factor (15 levels + 0)
506 float blendIndex = (index & 0x0F) / 16.0f;
507
508 // support for uint16_t
509 if constexpr (sizeof(UIntTy) > 1)
510 {
511 const float remapedIndex = (index / static_cast<float>(UINT16_MAX)) * 16.f;
512 renormIndex = std::min<uint8_t>(floorf(remapedIndex), 15);
513 blendIndex = remapedIndex - renormIndex;
514 static_assert(std::is_same_v<UIntTy, uint16_t>, "u8 or u16 allowed only");
515 }
516 else
517 {
518 static_assert(std::is_same_v<UIntTy, uint8_t>, "u8 or u16 allowed only");
519 }
520
521 if (renormIndex >= 16)
522 return 0;
523
524 // limit blend index
525 if (blendIndex < 0.0f)
526 blendIndex = 0.0f;
527 if (blendIndex > 1.0f)
528 blendIndex = 1.0f;
529 blendIndex = 1.0f - blendIndex;
530
531 const uint32_t entry = palette[renormIndex];
532
533 // convert to rgb
534 uint8_t red1 = (entry & Red) >> 16;
535 uint8_t green1 = (entry & Lime) >> 8;
536 uint8_t blue1 = entry & Blue;
537
538 // need to blend the palette
539 if (blendIndex > 0)
540 {
541 uint32_t nextColor = 0;
542 if (renormIndex == 15)
543 {
544 nextColor = PaletteLoops ? palette[0] : palette[15];
545 }
546 else
547 {
548 nextColor = palette[1 + renormIndex];
549 }
550
551 const float f1 = blendIndex;
552 const float f2 = 1.0 - blendIndex;
553
554 const uint8_t red2 = (nextColor >> 16) & 0x0000ff;
555 red1 = red1 * f1;
556 red1 += red2 * f2;
557
558 const uint8_t green2 = (nextColor >> 8) & 0x0000ff;
559 green1 = green1 * f1;
560 green1 += green2 * f2;
561
562 const uint8_t blue2 = nextColor & 0x0000ff;
563 blue1 = blue1 * f1;
564 blue1 += blue2 * f2;
565 }
566
567 if (brightness != 255)
568 {
569 if (brightness != 0)
570 {
571 const float adjustedBrightness = (brightness + 1) / 256.0; // adjust for rounding
572 // Now, since brightness is nonzero, we don't need the full scale8_video
573 // logic; we can just to scale8 and then add one (unless scale8 fixed) to
574 // all nonzero inputs.
575 if (red1)
576 {
577 red1 = red1 * adjustedBrightness;
578 ++red1;
579 }
580 if (green1)
581 {
582 green1 = green1 * adjustedBrightness;
583 ++green1;
584 }
585 if (blue1)
586 {
587 blue1 = blue1 * adjustedBrightness;
588 ++blue1;
589 }
590 }
591 else
592 {
593 red1 = 0;
594 green1 = 0;
595 blue1 = 0;
596 }
597 }
598
599 // return color code
600 uint32_t outputColor = red1;
601 outputColor = (outputColor << 8) | green1;
602 outputColor = (outputColor << 8) | blue1;
603 return outputColor;
604}
