24template<uint8_t freezeSize = 5, bool forwardRamp = true> void rampRainbow(auto& ctx, uint8_t rampValue)
25{
26 uint16_t overlaySize = (ctx.lamp.ledCount * rampValue) / 256;
27 for (uint16_t i = 0; i < overlaySize; ++i)
28 {
29 uint16_t colorAngle = (i * 360) / overlaySize + (forwardRamp ? rampValue * 4 : 0);
30 auto color = modes::colors::fromAngleHue(colorAngle);
31 ctx.lamp.setPixelColor(i, color);
32 }
33
34 for (uint16_t i = 0; i < 5; ++i)
35 {
36 if (i == 0 && overlaySize < ctx.lamp.ledCount)
37 {
38 ctx.lamp.setPixelColor(overlaySize + i, modes::colors::fromRGB(255, 255, 255));
39 continue;
40 }
41
42 if (overlaySize + i < ctx.lamp.ledCount)
43 {
44 ctx.lamp.setPixelColor(overlaySize + i, modes::colors::fromRGB(0, 0, 0));
45 }
46 }
47
48 ctx.skipNextFrames(freezeSize);
49}

67template<uint8_t freezeSize = 5, uint8_t nbBlackLines = 2>
68void rampColorRing(auto& ctx, uint8_t rampValue, auto palette)
69{
70 ctx.skipFirstLedsForFrames(0);
71
72 for (uint8_t i = 0; i < ctx.lamp.maxWidth * nbBlackLines; ++i)
73 {
74 ctx.lamp.setPixelColor(i, 0);
75 }
76
77 for (uint16_t i = 0; i < ctx.lamp.maxWidth; ++i)
78 {
79 uint8_t rampScale = (i * 255) / ctx.lamp.maxWidth;
80 if (rampScale < rampValue)
81 {
82 const auto color = modes::colors::from_palette(rampScale, palette);
83 ctx.lamp.setPixelColor(i, color);
84 }
85 else
86 {
87 ctx.lamp.setPixelColor(i, modes::colors::Black);
88 }
89 }
90
91 ctx.skipFirstLedsForFrames(ctx.lamp.maxWidth * nbBlackLines, freezeSize);
92}
