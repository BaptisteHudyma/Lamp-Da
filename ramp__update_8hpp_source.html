28template<uint8_t freezeSize = 5, bool forwardRamp = true> void rampRainbow(auto& ctx, uint8_t rampValue)
29{
30 uint16_t overlaySize = (ctx.lamp.ledCount * rampValue) / 256;
31 for (uint16_t i = 0; i < overlaySize; ++i)
32 {
33 uint16_t colorAngle = (i * 360) / overlaySize + (forwardRamp ? rampValue * 4 : 0);
34 auto color = modes::colors::fromAngleHue(colorAngle);
35 ctx.lamp.setPixelColor(i, color);
36 }
37
38 for (uint16_t i = 0; i < 5; ++i)
39 {
40 if (i == 0 && overlaySize < ctx.lamp.ledCount)
41 {
42 ctx.lamp.setPixelColor(overlaySize + i, modes::colors::fromRGB(255, 255, 255));
43 continue;
44 }
45
46 if (overlaySize + i < ctx.lamp.ledCount)
47 {
48 ctx.lamp.setPixelColor(overlaySize + i, modes::colors::fromRGB(0, 0, 0));
49 }
50 }
51
52 ctx.skipNextFrames(freezeSize);
53}
