21 static constexpr int nbBubbles = 4; // 0-10 max number of bubble per turn
22 static constexpr int bubbleFreq = 70; // 0-255 the more the less bubbles
23 static constexpr int minBubbleDist = 4; // 0-5 min distance of bubbles?
24 static constexpr int eventFreq = 70; // 0-255 how likely are events?
25 static constexpr int algeaFreq = 60; // 0-255 how likely event is algea?
26 static constexpr int algeaLength = 3000; // (ms) +/- 1s average algea size
27 static constexpr int algeaSwims = 400; // (ms) how fast algea ondulates
28 static constexpr int starFreq = 3; // 0-255 how likely event is star?

32 static constexpr auto algeaColor = colors::ForestGreen;
33 static constexpr auto starColor = colors::Gold;

36 {
37 // fill the initial automata with zeros +few colored pixels
38 GridTy::LineTy first {};
39 first.fill(waterColor);
40
41 // reset grid object & config
42 ctx.state.grid.reset(ctx.lamp, first);
43
44 // reset stateful events
45 ctx.state.algeaStart = 0;
46 }
47
48 static void loop(auto& ctx)
49 {
50 // rng helper
51 uint8_t mini = ctx.get_active_custom_ramp() / 4;
52 uint8_t maxi = 255;
53 auto rng = [&](auto p) LMBD_INLINE {
54 return random8(mini, maxi) < p;
55 };
56
57 // st & lamp for brievty
58 auto& st = ctx.state;
59 auto& lamp = ctx.lamp;
60
61 // callback to update the automaton
62 auto cb = [&](const auto& before, auto& after) {
63 after.fill(waterColor);
64
65 // add randomly bubbles
66 for (size_t I = 0; I < nbBubbles; ++I)
67 {
68 if (rng(bubbleFreq))
69 after[random8(0, after.size())] = colors::fromGrey(0xff);
70 }
71
72 // one pass to remove too crowded stuff
73 int lastBubble = 255;
74 for (int I = 0; I < after.size(); ++I)
75 {
76 // if something above, cancel bubble
77 if (before[I] != waterColor)
78 lastBubble = I;
79
80 if (after[I] == waterColor)
81 continue;
82
83 // if other bubble too close, cancel bubble
84 if (abs(int(lastBubble - I)) < minBubbleDist)
85 {
86 after[I] = waterColor;
87 }
88
89 lastBubble = I;
90 }
91
92 // trigger events
93 if (rng(eventFreq) && rng(eventFreq))
94 {
95 if (rng(starFreq))
96 {
97 after[random8(0, after.size())] = starColor;
98 }
99
100 if (st.algeaStart == 0 && rng(algeaFreq))
101 {
102 st.algeaPos = random8(2, after.size() - 2);
103 st.algeaStart = lamp.now;
104 st.algeaLifetime = random16(algeaLength - 1000, algeaLength + 1000);
105 }
106 }
107
108 // draw oscillating algea if active
109 if (st.algeaStart && st.algeaStart + st.algeaLifetime > lamp.now)
110 {
111 int shift = abs(int(((lamp.now / algeaSwims) % 8) - 3)) - 2; // oscillate -2..+2
112 after[st.algeaPos + shift] = algeaColor;
113 }
114 else
115 {
116 st.algeaStart = 0;
117 }
118 };
119
120 // default grid loop
121 ctx.state.grid.template loop<false>(ctx, cb);
122 }
123
124 static constexpr bool hasCustomRamp = true;
125};
