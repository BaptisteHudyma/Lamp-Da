55 {
56 // fill the initial automata with zeros +few colored pixels
57 GridTy::LineTy first {};
58 first.fill(waterColor);
59
60 // reset grid object & config
61 ctx.state.grid.reset(ctx.lamp, first);

64 ctx.state.algeaStart = 0;

67 ctx.template set_config_bool<ConfigKeys::rampSaturates>(true);

70 static void loop(auto& ctx)
71 {
72 // rng helper
73 uint8_t mini = ctx.get_active_custom_ramp() / 4;
74 uint8_t maxi = 255;
75 auto rng = [&](auto p) LMBD_INLINE {
76 return random8(mini, maxi) < p;
77 };
78
79 // st & lamp for brievty
80 auto& st = ctx.state;
81 auto& lamp = ctx.lamp;
82
83 // callback to update the automaton
84 auto cb = [&](const auto& before, auto& after) {
85 after.fill(waterColor);
86
87 // add randomly bubbles
88 for (size_t I = 0; I < nbBubbles; ++I)
89 {
90 if (rng(bubbleFreq))
91 after[random8(0, after.size())] = colors::fromGrey(0xff);
92 }
93
94 // one pass to remove too crowded stuff
95 int lastBubble = 255;
96 for (int I = 0; I < after.size(); ++I)
97 {
98 // if something above, cancel bubble
99 if (before[I] != waterColor)
100 lastBubble = I;
101
102 if (after[I] == waterColor)
103 continue;
104
105 // if other bubble too close, cancel bubble
106 if (abs(int(lastBubble - I)) < minBubbleDist)
107 {
108 after[I] = waterColor;
109 }
110
111 lastBubble = I;
112 }
113
114 // trigger events
115 if (rng(eventFreq) && rng(eventFreq))
116 {
117 if (rng(starFreq))
118 {
119 after[random8(0, after.size())] = starColor;
120 }
121
122 if (st.algeaStart == 0 && rng(algeaFreq))
123 {
124 st.algeaPos = random8(2, after.size() - 2);
125 st.algeaStart = lamp.now;
126 st.algeaLifetime = random16(algeaLength - 1000, algeaLength + 1000);
127 }
128 }
129
130 // draw oscillating algea if active
131 if (st.algeaStart && st.algeaStart + st.algeaLifetime > lamp.now)
132 {
133 int shift = abs(int(((lamp.now / algeaSwims) % 8) - 3)) - 2; // oscillate -2..+2
134 after[st.algeaPos + shift] = algeaColor;
135 }
136 else
137 {
138 st.algeaStart = 0;
139 }
140 };
141
142 // default grid loop
143 ctx.state.grid.template loop<false>(ctx, cb);
144 }
146 static constexpr bool hasCustomRamp = true;
147};
