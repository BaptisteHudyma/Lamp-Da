123static constexpr uint32_t indexKeyFlag = (1 << 31);

141enum class PrefixValues : uint32_t
142{
143 generalKeys = indexKeyFlag | (0x0 << 29),
144 managerKeys = indexKeyFlag | (0x1 << 29),
145 privateKeys = indexKeyFlag | (0x2 << 29),
146 invalidKeys = indexKeyFlag | (0x3 << 29),
147};
148
150static constexpr uint32_t noGroupIndex = 15;
151
153static constexpr uint32_t noModeIndex = 31;
154
156static constexpr uint32_t LMBD_INLINE
157indexKeyFor(PrefixValues prefix, uint32_t groupId, uint32_t modeId, uint32_t offset, uint32_t index)
158{
159 uint32_t rawPrefix = (uint32_t)prefix;
160
161 assert((rawPrefix >> 31) == 0b1); // prefix on 2 of the MSBs +indexKeyFlag
162 assert((groupId >> 4) == 0); // groupId on 4 bits (0-14)
163 assert((modeId >> 5) == 0); // modeId on 5 bits (0-30)
164 assert((offset >> 4) == 0); // offset on 4 bits (0-15)
165 assert((index >> 16) == 0); // index on 16 bits (0-65535)
166
167 return rawPrefix | (groupId << 25) | (modeId << 20) | (offset << 16) | index;
168}

171template<typename _EnumTy, int _groupId, int _modeId, PrefixValues _prefix = PrefixValues::generalKeys> struct KeyStore
172{
173 using EnumTy = _EnumTy;
174 static constexpr uint32_t groupId = _groupId < 0 ? noGroupIndex : _groupId;
175 static constexpr uint32_t modeId = _modeId < 0 ? noModeIndex : _modeId;

181 static_assert(sizeof(EnumTy) == sizeof(uint16_t), "uint16_t enum only");

185 // \private maps to ContextTy::KeyProxy::setValue()
186 template<EnumTy key, typename T> static inline void LMBD_INLINE setValue(T value)
187 {
188 using valueTy = std::remove_cv_t<std::remove_reference_t<T>>;
189 constexpr uint16_t rawKey = (uint16_t)key;
190
191 static_assert(storeIdKey != rawKey, "error: key can not match storeIdKey (0xafff)");
192
193 // simple case: value is uint32_t, store it as-is
194 if constexpr (std::is_same_v<valueTy, uint32_t>)
195 {
196 constexpr auto idx = indexKeyFor(prefix, groupId, modeId, 0, rawKey);
197 fileSystem::user::set_value(idx, value);
198
199 // small case: value is smaller than uint32_t, store it on a uint32_t
200 }
201 else if constexpr (sizeof(valueTy) <= sizeof(uint32_t))
202 {
203 uint32_t storage = 0;
204 details::bit_cast<sizeof(value)>(storage, value);
205 setValue<key>(storage);
206
207 // large case: value is stored on up to 16 uint32_t
208 }
209 else if constexpr (sizeof(valueTy) <= 16 * sizeof(uint32_t))
210 {
211 constexpr uint8_t N = ((sizeof(valueTy) - 1) / sizeof(uint32_t)) + 1;
212 std::array<uint32_t, N> storage {};
213 details::bit_cast<sizeof(value)>(storage, value);
214 for (uint8_t I = 0; I < N; ++I)
215 {
216 uint32_t off = indexKeyFor(prefix, groupId, modeId, I, rawKey);
217 fileSystem::user::set_value(off, storage[I]);
218 }
219 }
220 else
221 {
222 static_assert(sizeof(valueTy) <= 64, "KeyStore values limited to 64 bytes!");
223 }
224 }

227 template<EnumTy key, typename T> static inline bool LMBD_INLINE getValue(T& output)
228 {
229 constexpr uint16_t rawKey = (uint16_t)key;

231 static_assert(storeIdKey != rawKey, "error: key can not match storeIdKey (0xafff)");
232
233 // simple case: value is uint32_t, get it as-is
234 if constexpr (std::is_same_v<T, uint32_t>)
235 {

239 // small case: value is smaller than uint32_t, get it on a uint32_t
240 }
241 else if constexpr (sizeof(T) <= sizeof(uint32_t))
242 {

251
252 // large case: value is stored on up to 16 uint32_t
253 }
254 else if constexpr (sizeof(T) <= 16 * sizeof(uint32_t))
255 {
256 constexpr uint8_t N = ((sizeof(T) - 1) / sizeof(uint32_t)) + 1;
257 std::array<uint32_t, N> storage {};
258 for (uint8_t I = 0; I < N; ++I)
259 {
260 uint32_t off = indexKeyFor(prefix, groupId, modeId, I, rawKey);
261 if (not fileSystem::user::get_value(off, storage[I]))
262 {
263 return false;
264 }
265 }
266
267 details::bit_cast<sizeof(output)>(output, storage);
268 return true;
269 }
270 else
271 {
272 static_assert(sizeof(T) <= 64, "KeyStore values limited to 64 bytes!");
273 }

275 }

293 // \private empty store if \p storeId mismatch with value at \p storeIdKey
294 template<uint32_t storeId> static void migrateStoreIfNeeded()
295 {
296 constexpr auto idx = indexKeyFor(prefix, groupId, modeId, 0, storeIdKey);
297
298 // retrieve storeId from storage
299 uint32_t otherId = 0;
300 if (not fileSystem::user::get_value(idx, otherId))
301 {
302 otherId = storeId;
303 fileSystem::user::set_value(idx, storeId);
304 }
305
306 // if it don't match our storeId, clean storage from our old keys
307 if (otherId != storeId || otherId == 0)
308 {
309 constexpr uint32_t select = 0xfff00000; // all but offset
310 constexpr uint32_t masked = idx & select;
311 fileSystem::user::dropMatchingKeys(masked, select);
312 fileSystem::user::set_value(idx, storeId);
313 }

318template<uint32_t baseId, typename AllModes> static constexpr uint32_t derivateStoreIdImpl()
319{
320 constexpr uint32_t MixCstA = 0x8af8901d, MixCstB = 0x2f9c7c90;
321 constexpr uint8_t NbModes = std::tuple_size_v<AllModes>;
322
323 uint32_t storeId = baseId;
324 details::unroll<NbModes>([&](auto Idx) {
325 constexpr uint8_t I = decltype(Idx)::value;
326 using ModeHere = std::tuple_element_t<I, AllModes>;
327 using StoreHere = StoreEnumOf<ModeHere>;
328 constexpr bool hasStore = not std::is_same_v<StoreHere, NoStoreHere>;
329
330 uint32_t storeIdHere = 0;
331 if constexpr (hasStore)
332 {
333 storeIdHere = ModeHere::storeId;
334 }
335
336 // straightforward mix function
337 storeId += MixCstA;
338 storeId = (storeId << 7) ^ (storeId >> (32 - 7));
339 storeId ^= MixCstB;
340
341 storeId += storeIdHere;
342 });
343
344 return storeId;
345}
