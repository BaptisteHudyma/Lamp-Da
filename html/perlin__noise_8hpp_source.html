83 static int16_t get_next_speed(auto& ctx, const uint32_t position, int16_t speed)
84 {
85 static constexpr int16_t speedBleedof = 25;
86 static constexpr int16_t acceleration = 25;
87 static constexpr uint32_t confortZone = UINT32_MAX / 200000;
88
89 // close to zero, set mostly positive speeds
90 if (position < confortZone)
91 {
92 // bleed off speed to zero
93 if (speed < 0)
94 speed += speedBleedof;
95 else
96 speed = lmpd_constrain<int16_t>(
97 speed + lmpd_map<int16_t>(random8(), 0, 255, 0, acceleration * 2), 0, ctx.state.maxSpeed);
98 }
99 else if (position > (UINT32_MAX - confortZone))
100 {
101 // bleed off speed to zero
102 if (speed > 0)
103 speed -= speedBleedof;
104 else
105 speed = lmpd_constrain<int16_t>(
106 speed + lmpd_map<int16_t>(random8(), 0, 255, -acceleration * 2, 0), -ctx.state.maxSpeed, 0);
107 }
108 else
109 {
110 // free range x speed
111 speed = lmpd_constrain<int16_t>(speed + lmpd_map<int16_t>(random8(), 0, 255, -acceleration, acceleration),
112 -ctx.state.maxSpeed,
113 ctx.state.maxSpeed);
114 // prevent too slow speed
115 if (speed < 0 and speed > -ctx.state.minSpeed)
116 speed = -ctx.state.minSpeed;
117 else if (speed > 0 and speed < ctx.state.minSpeed)
118 speed = ctx.state.minSpeed;
119 }
120
121 return speed;
122 }

124 static void loop(auto& ctx)
125 {
126 auto& state = ctx.state;
127 auto& lamp = ctx.lamp;
128 if (!state.selectedPalette)
129 return;
130
131 // reference to a value buffer
132 static auto noiseBuffer = lamp.template getTempBuffer<bufferIndexToUse>();
133
134 // copy to prevent a ramp update mid animation
135 const colors::PaletteTy palette = *(state.selectedPalette);
136
137 const auto x = state.positionX;
138 const auto y = state.positionY;
139 const auto z = state.positionZ;
140 const auto scale = state.scale;
141
142 // this is too heavy to run at full, speed, display every other pixels instead of refreshing amm
143 static constexpr size_t everyNIndex = 2;
144 const size_t firstIndex = lamp.tick % everyNIndex;
145
146 // update noise values
147 for (size_t i = firstIndex; i < lamp.ledCount; i += everyNIndex)
148 {
149 const auto res = modes::strip_to_helix_unconstraint(i);
150 uint16_t data = noise16::inoise(x + scale * res.x, y + scale * res.y, z + scale * res.z);
151
152 // smooth over time to prevent suddent jumps
153 noiseBuffer[i] = data;
154 }
155
156 // apply slow drift to X and Y, just for visual variation.
157 state.positionX += state.speedX;
158 state.positionY += state.speedY;
159 state.positionZ += state.speedZ;
160
161 // vary speed
162 state.speedX = get_next_speed(ctx, state.positionX, state.speedX);
163 state.speedY = get_next_speed(ctx, state.positionY, state.speedY);
164 state.speedZ = get_next_speed(ctx, state.positionZ, state.speedZ);
165
166 for (size_t i = firstIndex; i < lamp.ledCount; i += everyNIndex)
167 {
168 uint16_t index = noiseBuffer[i];
169 uint8_t bri = noiseBuffer[lamp.ledCount - 1 - i] >> 8;
170
171 // if this palette is a 'loop', add a slowly-changing base value
172 index += state.ihue;
173
174 // brighten up, as the color palette itself often contains the
175 // light/dark dynamic range desired
176 if (bri > 127)
177 {
178 bri = 255;
179 }
180 else
181 {
182 bri = dim8_raw(bri * 2);
183 }
184
185 lamp.setPixelColor(i, colors::from_palette(index, palette, bri));
186 }
187
188 state.ihue += 1;
189 }
