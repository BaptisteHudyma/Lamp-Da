218 uint8_t LMBD_INLINE set_active_mode(uint8_t value, uint8_t maxValueWrap = 255)
219 {
220 if (value + 1 > maxValueWrap)
221 value = 0;
222
223 auto manager = modeManager.get_context();
224
225 // cancel any blip
226 // TODO: this should be in quit_mode
227 manager.cancel_blip();
228
229 // if next mode is the same as current one, add a blip to help user differenciate
230 if (modeManager.activeIndex.modeIndex == value)
231 {
232 // This is not complete, there is no group check, so the blip will be visible sometimes on modes with the same
233 // modeIndex in different groups
234
235 // blip to indicate a mode change to the same mode
236 manager.blip(100);
237 }
238
239 // signal that we are quitting the mode
240 modeManager.quit_mode(manager);
241 // switch mode
242 modeManager.activeIndex.modeIndex = value;
243 // signal that we entered a new mode
244 modeManager.enter_mode(manager);
245 return value;
246 }
