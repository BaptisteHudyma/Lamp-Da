111 void iterate_with_collisions(const vec3d& accelerationCartesian,
112 const float deltaTime_s,
113 const bool shouldContrain = true)
114 {
115 for (size_t i = 0; i < particuleCount; ++i)
116 {
117 // do not iterate non allocated particles
118 if (not isAllocated[i])
119 continue;
120
121 Particle& p = particules[i];
122 const int16_t ledIndex = p._savedLampIndex;
123
124 // simulate instead of updating directly
125 const Particle& newP = p.simulate_after_acceleration(accelerationCartesian, deltaTime_s, shouldContrain);
126
127 // update particle position in occupation set
128 const int16_t newLedIndex = newP._savedLampIndex;
129 if (newLedIndex != ledIndex)
130 {
131 // check collision : no collision
132 if (not is_position_taken(newLedIndex))
133 {
134 occupiedSpacesSet.erase(ledIndex);
135 occupiedSpacesSet.insert(newLedIndex);
136 }
137 // check collision : collision !!
138 else
139 {
140 // refuse movement, rebound speed
141 p.thetaSpeed_radS = -p.thetaSpeed_radS * 0.75;
142 p.zSpeed_mS = -p.zSpeed_mS * 0.75;
143 continue;
144 }
145 }
146
147 // update particle
148 p = newP;
149 }
