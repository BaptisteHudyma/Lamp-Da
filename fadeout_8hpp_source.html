71 void update_depop_rate(auto& ctx, const float progress)
72 {
73 const float updateDuration = static_cast<float>((ctx.lamp.now - latestProgressTime) / 1000.0);
74 const float progressPerSecond = (progress - latestProgress) / updateDuration;
75
76 bool isValidCall = latestProgress >= 0 and updateDuration > 0;
77
78 // cancel the particle spawn
79 if (progress <= 0.0)
80 {
81 particlesToDepopPerIteration = 0.0;
82 latestProgress = -1;
83 isValidCall = false;
84 }
85 // update progress
86 else if (latestProgress != progress)
87 {
88 latestProgress = progress;
89 latestProgressTime = ctx.lamp.now;
90 }
91 else
92 isValidCall = false;
93
94 // not valid call, skip update
95 if (not isValidCall)
96 {
97 return;
98 }
99
100 const uint16_t particleMax = ctx.lamp.ledCount;
101 const float FramesFrequency = static_cast<float>(ctx.lamp.frameDurationMs / 1000.0);
102
103 const float particlesLeft = particleMax - particlesDropped;
104 const float trueProgress = 1.0 - particlesLeft / static_cast<float>(particleMax);
105 const float trueProgressDiff = progress - trueProgress;
106 // try to correct the update ratio by the lag or advance
107 const float correctedParticlesLeft = particleMax * (1.0 + trueProgressDiff);
108
109 // particles to depop per second is progress rate per second * particles left
110 particlesToDepopPerIteration = progressPerSecond * correctedParticlesLeft * FramesFrequency;
111 }

120 bool depop_one_particles(auto& ctx)
121 {
122 bool isDepoped = false;
123 // progress goes from 0 to 1
124 // lamp turns off when 1 is reached
125 auto& buffer = ctx.lamp.template getTempBuffer<MaskBuffId>();
126
127 // check lines by lines, keeping the first one with pixels left
128 for (int16_t y = ctx.lamp.maxHeight; y >= 0; y--)
129 {
130 // search for the first line with pixels sets, from bottom to top
131 uint8_t pixelSetCount = 0;
132 for (size_t x = 0; x <= ctx.lamp.maxWidth; x++)
133 {
134 const bool isPixelSet = buffer[modes::to_strip(x, y)] != 0;
135 // check that at leat one pixel is still set
136 if (isPixelSet)
137 pixelSetCount += 1;
138 }
139
140 // found the first line with pixels left
141 if (pixelSetCount > 0)
142 {
143 // get a random led index in this
144 uint8_t selectedLed = lmpd_map<uint8_t>(rand(), 0, RAND_MAX, 0, pixelSetCount);
145 for (size_t x = 0; x <= ctx.lamp.maxWidth; x++)
146 {
147 const auto& pixelId = modes::to_strip(x, y);
148 const bool isPixelSet = buffer[pixelId] != 0;
149 if (isPixelSet)
150 {
151 if (selectedLed == 0)
152 {
153 // drop this pixel
154 buffer[pixelId] = 0;
155 particlesDropped += 1;
156 // spawn a new particle
157 particuleSystem.init_deferred_particules(1, [pixelId](size_t) {
158 return pixelId;
159 });
160
161 isDepoped = true;
162 break;
163 }
164 selectedLed -= 1;
165 }
166 }
167 break;
168 }
169 }
170 if (not isDepoped)
171 {
172 platform::lampda_print("Error: Could not depop particle");
173 return false;
174 }
175 return true;
176 }
