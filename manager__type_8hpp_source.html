83template<bool displayFavoriteNumber = true> void _animate_favorite_pick(auto& ctx, float holdDuration, float stepSize)
84{
85 // user as a number of favorite set
86 // occasional +1 if not all favorite are set (allow a new favorite)
87 const uint8_t numberOfFavoriteSet =
88 ctx.state.usedFavoriteCount + ((ctx.state.usedFavoriteCount < ctx.state.maxFavoriteCount) ? 1 : 0);
89
90 // up to maxFavoriteCount step state: "which_one" is [0, 1, 2, 3, ...] and "do not set" is the max index + 1
91 uint32_t stepCount = numberOfFavoriteSet + floor(holdDuration / stepSize);
92 stepCount = stepCount % (numberOfFavoriteSet + 1);
93
94 // display ramp to show where user is standing
95 if (stepCount >= numberOfFavoriteSet)
96 {
97 // cancel action
98 ctx.state.isFavoritePending = 0;
99
100 // display ramp for the first time to allow the user to cancel the action
101 if (holdDuration <= 2 * stepSize)
102 {
103 _animate_ramp(ctx, holdDuration, stepSize, colors::PaletteGradient<colors::White, colors::Cyan>);
104 }
105 }
106 else
107 {
108 // animate with different colors
109 switch (stepCount % 4)
110 {
111 case 0:
112 _animate_ramp(ctx, holdDuration, stepSize, colors::PaletteGradient<colors::Green, colors::White>);
113 break;
114 case 1:
115 _animate_ramp(ctx, holdDuration, stepSize, colors::PaletteGradient<colors::Blue, colors::White>);
116 break;
117 case 2:
118 _animate_ramp(ctx, holdDuration, stepSize, colors::PaletteGradient<colors::Orange, colors::White>);
119 break;
120 case 3:
121 _animate_ramp(ctx, holdDuration, stepSize, colors::PaletteGradient<colors::Purple, colors::White>);
122 break;
123 }
124
125 // extra display on the first pixels (count pixels to know fav no)
126 if constexpr (displayFavoriteNumber)
127 {
128 // display ramp
129 display_favorite_number_ramp(ctx, stepCount, numberOfFavoriteSet, stepCount < numberOfFavoriteSet);

244 void LMBD_INLINE update_ramp(uint8_t rampValue, uint32_t holdTime, auto callback)
245 {
246 // restart the rampage
247 if (holdTime < lastTimeMeasured)
248 {
249 lastTimeMeasured = holdTime;
250
251 if (holdTime < startPeriod)
252 {
253 // toggle forward / backward direction
254 isForward = !isForward;
255 if (rampSaturates)
256 {
257 if (rampValue == 0)
258 isForward = true;
259 if (rampValue == 255)
260 isForward = false;
261 }
262 }
263 return;
264 }
265
266 // count how many step we advanced
267 uint32_t lastCounter = lastTimeMeasured / stepSpeed;
268 uint32_t nextCounter = holdTime / stepSpeed;
269 if (nextCounter <= lastCounter)
270 return;
271
272 // apply steps
273 for (uint32_t I = 0; I < nextCounter - lastCounter; ++I)
274 {
275 // increment iff possible
276 if (isForward && (!rampSaturates || rampValue < 255))
277 rampValue += 1;
278
279 // decrement iff possible
280 if (!isForward && (!rampSaturates || rampValue > 0))
281 rampValue -= 1;
282
283 // forward value to callback
284 callback(rampValue);
285 }
286
287 lastTimeMeasured = holdTime;
288 }
289
290 uint32_t stepSpeed;
291 bool rampSaturates;
292 bool animEffect;
293 bool animChoice;
294 uint32_t lastTimeMeasured;
295 bool isForward;

491 {

593 static bool jump_to_favorite(auto& ctx, uint8_t which_one, bool shouldSaveLastActiveIndex)
594 {
595 // sanity check
596 if (ctx.state.usedFavoriteCount <= 0)
597 return false;
598
599 // wrap back to max number of favorites
600 which_one = (which_one % ctx.state.usedFavoriteCount);
601 ctx.state.lastFavoriteStep = which_one;
602
603 // store last active index before jump
604 if (shouldSaveLastActiveIndex)
605 {
606 ctx.state.beforeFavoriteGroupIndex = ctx.modeManager.activeIndex.groupIndex;
607 ctx.state.beforeFavoriteModeIndex = ctx.modeManager.activeIndex.modeIndex;
608 }
609

611 return false;
612
613 const auto targetFavorite = ctx.state.favorites[which_one];
614 // reset once with the right mode
615 jump_to_new_active_index(ctx, targetFavorite);

648 {
649 // delete current
650 const auto which_one = ctx.state.lastFavoriteStep;
651 if (ctx.state.usedFavoriteCount > 0 and which_one < ctx.state.maxFavoriteCount)

688 static void handle_scroll_modes(auto& ctx, uint32_t holdDuration)
689 {
690 auto& scrollHandler = ctx.state.scrollHandler;
691 scrollHandler.isForward = false; // (always scroll modes backward)
692
693 static constexpr uint32_t scrollActivationTiming = 750;
694 if (holdDuration <= scrollActivationTiming)
695 {
696 // display the ramp and do nothing else
697 modes::details::_animate_ramp(

699 return;

702 scrollHandler.update_ramp(128, holdDuration, [&](uint8_t rampValue) {
703 uint8_t modeIndex = ctx.get_active_mode();
704 uint8_t groupIndex = ctx.get_active_group();
705 uint8_t modeCount = ctx.get_modes_count();
706 uint8_t groupCount = ctx.get_groups_count();
707
708 ctx.state.isLastScrollAGroupChange = false;
709
710 // we are going backward
711 //
712 if (rampValue < 128)
713 {
714 // if modeIndex is not the first, just decrement it
715 if (modeIndex > 0)
716 {
717 ctx.set_active_mode(modeIndex - 1, modeCount);
718
719 // or else decrement group, then set mode to last one
720 }
721 else
722 {
723 ctx.state.isLastScrollAGroupChange = true;
724 // if groupIndex is not the first, just decrement it
725 if (groupIndex > 0)
726 {
727 ctx.set_active_group(groupIndex - 1, groupCount);
728
729 // else wrap to last group
730 }
731 else
732 {
733 ctx.set_active_group(groupCount - 1, groupCount);
734 }
735
736 // backward scroll: set mode to last one on group change
737 modeCount = ctx.get_modes_count();
738 ctx.set_active_mode(modeCount - 1, modeCount);
739 }
740
741 // we are going forward
742 //
743 }
744 else
745 {
746 // if modeIndex is not the last, just increment it
747 if (modeIndex + 1 < modeCount)
748 {
749 ctx.next_mode();
750
751 // or else increment group
752 }
753 else
754 {
755 ctx.state.isLastScrollAGroupChange = true;
756 // if groupIndex is not the last, just increment it
757 if (groupIndex + 1 < groupCount)
758 {
759 ctx.next_group();
760
761 // else wrap to first group
762 }
763 else
764 {
765 ctx.set_active_group(0, groupCount);
766 }
767
768 // forward scroll: set mode to first one on group change
769 ctx.set_active_mode(0, modeCount);
770 }
771 }
772 });
