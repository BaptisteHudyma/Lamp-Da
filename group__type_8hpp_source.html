153 template<typename Mode> static auto* LMBD_INLINE getStateOf(auto& manager)
154 {
155 using StateTy = typename Mode::StateTy;
156 using OptionalTy = std::optional<StateTy>;
157
158 StateTy* substate = nullptr;
159 details::unroll<nbModes>([&](auto Idx) LMBD_INLINE {
160 using ModeHere = ModeAt<Idx>;
161 constexpr bool isHere = std::is_same_v<ModeHere, Mode>;
162
163 if constexpr (isHere)
164 {
165 auto* state = manager.template getStateGroupOf<SelfTy>();
166 if (state)
167 {
168 OptionalTy& opt = std::get<OptionalTy>(state->modeStates);
169 if (!opt.has_value())
170 {
171 opt.emplace(); // all StateTy must be default-contructible :)
172 }
173
174 StateTy& stateHere = *opt;
175 substate = &stateHere;
176 }
177 }
178 });
179
180 assert(substate != nullptr && "this should not have happened!");
181 return substate;
182 }
