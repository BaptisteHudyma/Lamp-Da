141 void update(auto& ctx)
142 {
143 const microphone::SoundStruct& soundObject = ctx.lamp.get_sound_struct();
144
145 // copy microphone data
146 data = soundObject.data;
147 dataAutoGained = soundObject.rectifiedData;
148 fft_log = soundObject.fft_log;
149 fft_raw = soundObject.fft_raw;
150 fft_log_end_frequencies = soundObject.fft_log_end_frequencies;
151 maxAmplitude = soundObject.maxAmplitude;
152 maxAmplitudeFrequency = soundObject.maxAmplitudeFrequency;
153
154 // average input sound over a second-long window (approx)
155 const auto soundLevel = soundObject.sound_level_Db;
156 level = (not std::isinf(soundLevel) and not std::isnan(soundLevel) and soundLevel > -70) ? soundLevel : -70;
157 avgLevel = (level + avgLevel * _windowSize) / (_windowSize + 1);
158
159 // average maximum over the same long window (approx)
160 avgMax = (avgLevel + max<float>(level, avgMax) * _windowSize) / (_windowSize + 1.0);
161 avgMax = max<float>(_eventCutoff, avgMax);
162
163 // delta between instantaneous level & average, squashed by its max
164 delta = max<float>(level - avgLevel, 0) / avgMax;
165
166 // average of delta square (1.0 split around _eventCutoff)
167 avgDelta = (delta * delta + avgDelta * _windowShort) / (_windowShort + 1.0);
168
169 // if avgDelta looks triggered, start sampling event
170 if (avgDelta > 1.0)
171 {
172 _eventCount += 1;
173 _eventScale = max<float>(_eventScale, 256.0 * (avgMax / _eventNorm) * (1.0 + delta));
174
175 // if no trigger, decay event
176 }
177 else
178 {
179 _eventCount = max<float>(1, _eventCount - 1);
180 }
181
182 // if event is active, and event decayed, reset eventScale & hasEvent
183 if (hasEvent)
184 {
185 _eventCount = 0;
186 _eventScale = 0;
187 eventScale = 0;
188 hasEvent = false;
189 }
190
191 // if no event is active, and enough samples, set eventScale & hasEvent
192 if (!hasEvent && _eventCount > nbEventSample)
193 {
194 eventScale = _eventScale;
195 hasEvent = true;
196 }
197
198 // beat tracking is heavy on performances
199 if constexpr (ConfigTy::useBeatTracking)
200 {
201 // update history size
202 if (historySize < _FFThistory_MaxSize)
203 historySize++;
204 else
205 beatDetected = track_beat_events(_FFTHistory_beatDetector, fft_log);
206
207 // add sample to history
208 _FFTHistory_beatDetector[historyIndex] = fft_log;
209 historyIndex++;
210 if (historyIndex >= _FFThistory_MaxSize)
211 historyIndex = 0;

216 bool is_beat_on_freq_range(const float minFreq, const float maxFreq)
217 {

244 }

281 template<size_t T> static inline std::array<bool, T> track_beat_events(const FFTHistoryContainer& dataHistory,
282 const std::array<float, T>& data)
283 {
284 std::array<bool, T> beats;
285 beats.fill(false);
286
287 // beat detection starts after a bit
288 const size_t dataHistoryCnt = dataHistory.size();
289 const float oneOverdataHistory = 1.0f / dataHistoryCnt;
290
291 std::array<float, T> averageFft;
292 averageFft.fill(0.0f);
293 for (const auto& fft: dataHistory)
294 {
295 for (size_t i = 0; i < T; i++)
296 averageFft[i] += fft[i] * oneOverdataHistory;
297 }
298
299 std::array<float, T> varianceFft;
300 varianceFft.fill(0.0f);
301 for (const auto& fft: dataHistory)
302 {
303 for (size_t i = 0; i < T; i++)
304 {
305 const float val = fft[i] - averageFft[i];
306 varianceFft[i] += val * val;
307 }
308 }
309
310 for (size_t i = 0; i < T; i++)
311 {
312 varianceFft[i] *= oneOverdataHistory;
313 const float stdDev = sqrtf(varianceFft[i]);
314 static constexpr float N = 1.5f;
315 // beat if > med + N * variance
316 // N=1 : greater than 68.0% of values
317 // N=2 : greater than 95.4% of values
318 // N=3 : greater than 99.6% of values
319 // N=4 : greater than 99.8% of values
320 beats[i] = data[i] > (averageFft[i] + N * stdDev);
321 }
322
323 return beats;
324 }
