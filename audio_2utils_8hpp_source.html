138 void update(auto& ctx)
139 {
140 const microphone::SoundStruct& soundObject = ctx.lamp.get_sound_struct();
141
142 // copy microphone data
143 data = soundObject.data;
144 dataAutoGained = soundObject.rectifiedData;
145 fft_log = soundObject.fft_log;
146 fft_raw = soundObject.fft_raw;
147 fft_log_end_frequencies = soundObject.fft_log_end_frequencies;
148 maxAmplitude = soundObject.maxAmplitude;
149 maxAmplitudeFrequency = soundObject.maxAmplitudeFrequency;
150
151 // average input sound over a second-long window (approx)
152 const auto soundLevel = soundObject.sound_level_Db;
153 level = (not std::isinf(soundLevel) and not std::isnan(soundLevel) and soundLevel > -70) ? soundLevel : -70;
154 avgLevel = (level + avgLevel * _windowSize) / (_windowSize + 1);
155
156 // average maximum over the same long window (approx)
157 avgMax = (avgLevel + max<float>(level, avgMax) * _windowSize) / (_windowSize + 1.0);
158 avgMax = max<float>(_eventCutoff, avgMax);
159
160 // delta between instantaneous level & average, squashed by its max
161 delta = max<float>(level - avgLevel, 0) / avgMax;
162
163 // average of delta square (1.0 split around _eventCutoff)
164 avgDelta = (delta * delta + avgDelta * _windowShort) / (_windowShort + 1.0);
165
166 // if avgDelta looks triggered, start sampling event
167 if (avgDelta > 1.0)
168 {
169 _eventCount += 1;
170 _eventScale = max<float>(_eventScale, 256.0 * (avgMax / _eventNorm) * (1.0 + delta));
171
172 // if no trigger, decay event
173 }
174 else
175 {
176 _eventCount = max<float>(1, _eventCount - 1);
177 }
178
179 // if event is active, and event decayed, reset eventScale & hasEvent
180 if (hasEvent)
181 {
182 _eventCount = 0;
183 _eventScale = 0;
184 eventScale = 0;
185 hasEvent = false;
186 }
187
188 // if no event is active, and enough samples, set eventScale & hasEvent
189 if (!hasEvent && _eventCount > nbEventSample)
190 {
191 eventScale = _eventScale;
192 hasEvent = true;
193 }
194

213 bool is_beat_on_freq_range(const float minFreq, const float maxFreq)
214 {

224 // climb to min freq bin

226 {

228 if (minFreq < maxFrequencyForBin)

231 // register beat events

239 if (maxFrequencyForBin >= maxFreq)
240 break;
241 }

278 template<size_t T> static inline std::array<bool, T> track_beat_events(const FFTHistoryContainer& dataHistory,
279 const std::array<float, T>& data)
280 {
281 std::array<bool, T> beats;
282 beats.fill(false);
283
284 // beat detection starts after a bit
285 const size_t dataHistoryCnt = dataHistory.size();
286 const float oneOverdataHistory = 1.0f / dataHistoryCnt;
287
288 std::array<float, T> averageFft;
289 averageFft.fill(0.0f);
290 for (const auto& fft: dataHistory)
291 {
292 for (size_t i = 0; i < T; i++)
293 averageFft[i] += fft[i] * oneOverdataHistory;
294 }
295
296 std::array<float, T> varianceFft;
297 varianceFft.fill(0.0f);
298 for (const auto& fft: dataHistory)
299 {
300 for (size_t i = 0; i < T; i++)
301 {
302 const float val = fft[i] - averageFft[i];
303 varianceFft[i] += val * val;
304 }
305 }
306
307 for (size_t i = 0; i < T; i++)
308 {
309 varianceFft[i] *= oneOverdataHistory;
310 const float stdDev = sqrtf(varianceFft[i]);
311 static constexpr float N = 1.5f;
312 // beat if > med + N * variance
313 // N=1 : greater than 68.0% of values
314 // N=2 : greater than 95.4% of values
315 // N=3 : greater than 99.6% of values
316 // N=4 : greater than 99.8% of values
317 beats[i] = data[i] > (averageFft[i] + N * stdDev);
318 }
319
320 return beats;
321 }
