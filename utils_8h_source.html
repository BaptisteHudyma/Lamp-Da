77template<typename T = float>
78static inline T lmpd_map(float x, const float in_min, const float in_max, const float out_min, const float out_max)
79{
80 using numeric_limits_T = std::numeric_limits<T>;
81 assert(not(std::isnan(in_min) or std::isinf(in_min)) && "in_min invalid");
82 assert(not(std::isnan(in_max) or std::isinf(in_max)) && "in_max invalid");
83 assert(out_min >= numeric_limits_T::lowest() && out_min <= numeric_limits_T::max() && "out_min invalid");
84 assert(not(std::isnan(out_min) or std::isinf(out_min)) && "out_min invalid");
85 assert(out_max >= numeric_limits_T::lowest() && out_max <= numeric_limits_T::max() && "out_max invalid");
86 assert(not(std::isnan(out_max) or std::isinf(out_max)) && "out_max invalid");
87
88 if (std::isnan(x))
89 return out_min;
90 if (x > 0 && std::isinf(x))
91 return out_max;
92 if (std::isinf(x))
93 return out_min;
94 const float res = (out_max - out_min) * (x - in_min) / (in_max - in_min) + out_min;
95 if (res > numeric_limits_T::max())
96 return numeric_limits_T::max();
97 if (res < numeric_limits_T::lowest())
98 return numeric_limits_T::lowest();
99
100 return static_cast<T>(res);
101}

173template<typename T> static constexpr uint32_t hash(const T s, const uint16_t maxSize = 14, const uint16_t off = 0)
174{
175#ifdef LMBD_CPP17
176 uint32_t hashAcc = 5381;
177 for (uint16_t I = 0; I < maxSize; ++I)
178 {
179 if (s[I + off] == '\0')
180 {
181 return hashAcc;
182 }
183
184 hashAcc = ((hashAcc << 5) + hashAcc) ^ s[I + off];
185 }
186 return hashAcc;
187#else
188 return !s[off] ? 5381 : (hash(s, maxSize, off + 1) * 33) ^ s[off];
189#endif
190}
