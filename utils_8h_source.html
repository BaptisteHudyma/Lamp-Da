44{
45#ifdef LMBD_CPP17

56template<typename T = float>
57static inline T lmpd_map(float x, const float in_min, const float in_max, const float out_min, const float out_max)
58{
59 using numeric_limits_T = std::numeric_limits<T>;
60 assert(not(std::isnan(in_min) or std::isinf(in_min)) && "in_min invalid");
61 assert(not(std::isnan(in_max) or std::isinf(in_max)) && "in_max invalid");
62 assert(out_min >= numeric_limits_T::lowest() && out_min <= numeric_limits_T::max() && "out_min invalid");
63 assert(not(std::isnan(out_min) or std::isinf(out_min)) && "out_min invalid");
64 assert(out_max >= numeric_limits_T::lowest() && out_max <= numeric_limits_T::max() && "out_max invalid");
65 assert(not(std::isnan(out_max) or std::isinf(out_max)) && "out_max invalid");
66
67 if (std::isnan(x))
68 return out_min;
69 if (x > 0 && std::isinf(x))
70 return out_max;
71 if (std::isinf(x))
72 return out_min;
73 const float res = (out_max - out_min) * (x - in_min) / (in_max - in_min) + out_min;
74 if (res > numeric_limits_T::max())
75 return numeric_limits_T::max();
76 if (res < numeric_limits_T::lowest())
77 return numeric_limits_T::lowest();
78
79 return static_cast<T>(res);
80}

150template<typename T> static constexpr uint32_t hash(const T s, const uint16_t maxSize = 14, const uint16_t off = 0)
151{
152#ifdef LMBD_CPP17
153 uint32_t hashAcc = 5381;
154 for (uint16_t I = 0; I < maxSize; ++I)
155 {
156 if (s[I + off] == '\0')
157 {
158 return hashAcc;
159 }
160
161 hashAcc = ((hashAcc << 5) + hashAcc) ^ s[I + off];
162 }
163 return hashAcc;
164#else
165 return !s[off] ? 5381 : (hash(s, maxSize, off + 1) * 33) ^ s[off];
166#endif
167}
