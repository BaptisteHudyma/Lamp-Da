171bool button_clicked(const uint8_t clicks)

173 auto manager = get_context();

175 switch (clicks)

177 case 6: // 6 clicks: jump to first mode of first category

196bool button_hold(const uint8_t clicks, const bool isEndOfHoldEvent, const uint32_t holdDuration)
197{
198 auto manager = get_context();
199 auto& rampHandler = manager.state.rampHandler;
200
201 switch (clicks)
202 {
203 case 3: // 3 click+hold: configure custom ramp
204 // no ramps in favorite group
205 if (not manager.state.isInFavoriteMockGroup)
206 {
207 rampHandler.update_ramp(manager.get_active_custom_ramp(), holdDuration, [&](uint8_t rampValue) {
208 manager.custom_ramp_update(rampValue);
209 manager.set_active_custom_ramp(rampValue);
210 });
211 return true;
212 }
213 break;
214
215 // 5 click+hold: Add 5 minutes to sunset timer
216 case 5:
217 {
218 if (not isEndOfHoldEvent and holdDuration > 0 and logic::sunset::is_enabled())
219 {
220 // sunset timer !
221 auto manager = get_context();
222 if (manager.overlay_animate_ramp(
223 holdDuration, 1000, modes::colors::PaletteGradient<modes::colors::White, modes::colors::Red>))
224 {
225 // update the sunset timing
226 manager.state.isSunsetTimingPending = 2;
227 }
228 }
229 break;
230 }
231
232 case 13: // 13 clicks + hold: reset the whole system and stored parameters
233 {
234 if (not isEndOfHoldEvent and holdDuration > 0)
235 {
236 auto manager = get_context();
237 if (manager.overlay_animate_ramp(
238 holdDuration, 5000, modes::colors::PaletteGradient<modes::colors::Red, modes::colors::Red>))
239 {
240 // reset the file system and memory
241 platform::lampda_print("clearing the whole file format");
242 physical::fileSystem::clear_internal_fs();
243
244 // shutdown the lamp
245 const bool shouldSaveUserParameters = false;
246 logic::behavior::internal::handle_shutdown_state(shouldSaveUserParameters);
247 }
248 }
249 break;
250 }
251
252 case 20: // 20 clicks + hold: reset the whole system and stored parameters
253 {
254 if (not isEndOfHoldEvent and holdDuration > 0)
255 {
256 auto manager = get_context();
257 if (manager.overlay_animate_ramp(
258 holdDuration, 5000, modes::colors::PaletteGradient<modes::colors::Red, modes::colors::Red>))
259 {
260 // reset the file system and memory
261 platform::lampda_print("clearing the whole file format");
262 physical::fileSystem::clear_internal_fs();
263
264 // shutdown the lamp
265 const bool shouldSaveUserParameters = false;
266 const bool shouldSaveSystemParameters = false;
267 logic::behavior::internal::handle_shutdown_state(shouldSaveUserParameters, shouldSaveSystemParameters);
268 }
269 }
270 break;
271 }
272 }
273
274 return false;
