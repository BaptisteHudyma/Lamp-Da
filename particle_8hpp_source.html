60 vec2d compute_speed_increment(const vec3d& accelerationCartesian_m, const float deltaTime_s) const
61 {
62 // speed vector on radial (derivative of cartesian to cylinder coordinates for theta)
63 const vec3d e_theta(-sin_t(theta_rad) * cylinderRadius_m, cos_t(theta_rad) * cylinderRadius_m, 0);
64 // speed vector on z (derivative of cartesian to cylinder coordinates for z)
65 const vec3d e_z(0, 0, 1);
66 // ignore the radius derivative, as we want to stay on the cylinder surface
67
68 // compute the speed vector on cylinder surface
69 const vec3d& tangantialVector = e_theta.multiply(accelerationCartesian_m.dot(e_theta));
70 const vec3d& directVector = e_z.multiply(accelerationCartesian_m.dot(e_z));
71 const vec3d& accelerationVector = tangantialVector.add(directVector);
72
73 return vec2d(angularSpeedGain * accelerationVector.dot(e_theta) / static_cast<float>(cylinderRadius_m) *
74 deltaTime_s,
75 linearSpeedGain * accelerationVector.dot(e_z) * deltaTime_s);
76 }

81 void apply_acceleration(const vec3d& accelerationCartesian_m,
82 const float deltaTime_s,
83 const bool shouldContrain = true)
84 {
85 const auto& speedIncrement = compute_speed_increment(accelerationCartesian_m, deltaTime_s);
86 // start by dampening the speed
87 dampen_speed(speedDampening);
88
89 // update speed
90 thetaSpeed_radS =
91 lmpd_constrain<float>(thetaSpeed_radS + speedIncrement.x, -maxAngularSpeed_radS, maxAngularSpeed_radS);
92 zSpeed_mS = lmpd_constrain<float>(zSpeed_mS + speedIncrement.y, -maxVerticalSpeed_mmS, maxVerticalSpeed_mmS);
93
94 static constexpr float angularUnit = LampTy::maxWidthFloat / c_TWO_PI;
95 static constexpr float verticalUnit = LampTy::ledStripWidth_mm * 1.5f;
96
97 // update position (limit speed to pixel unit per dt)
98 const float angularPositionIncrement =
99 lmpd_constrain<float>(thetaSpeed_radS * deltaTime_s, -angularUnit, angularUnit);
100 const float verticalPositionIncrement =
101 lmpd_constrain<float>(zSpeed_mS * deltaTime_s, -verticalUnit, verticalUnit) * 1000.0;
102
103 // update particle position
104 z_mm += verticalPositionIncrement;
105 theta_rad += angularPositionIncrement;
106
107 // limit the derivation of the angle
108 theta_rad = wrap_angle(theta_rad);
109
110 // constrain to the lamp body
111 if (shouldContrain)
112 constraint_into_lamp_body();
113 else
114 _savedLampIndex = to_lamp_index_no_bounds();
115 }

129 void constraint_into_lamp_body()
130 {
131 _savedLampIndex = to_lamp_index_no_bounds();
132
133 // coordinates go from 0 to -max
134 if (z_mm > 0)
135 {
136 z_mm = 0;
137 zSpeed_mS = -zSpeed_mS * reboundCoeff;
138 _savedLampIndex = to_lamp_index_no_bounds();
139 }
140
141 if (z_mm < -LampTy::lampHeight_mm)
142 {
143 z_mm = -LampTy::lampHeight_mm;
144 zSpeed_mS = -zSpeed_mS * reboundCoeff;
145 _savedLampIndex = to_lamp_index_no_bounds();
146 }
147
148 // handle the real limits (led strip do not start and end at zero depth)
149 if (not is_led_index_valid(_savedLampIndex))
150 {
151 // we are too high above the first led
152 if (z_mm >= 0)
153 {
154 // go one unit lower
155 z_mm = LampTy::ledStripWidth_mm * 1.5;
156 zSpeed_mS = -zSpeed_mS * reboundCoeff;
157 }
158 else
159 {
160 // go one unit above
161 z_mm = -LampTy::lampHeight_mm + LampTy::ledStripWidth_mm * 1.5;
162 zSpeed_mS = -zSpeed_mS * reboundCoeff;
163 }
164 // udpate index
165 _savedLampIndex = to_lamp_index_no_bounds();
166 }
167 }
